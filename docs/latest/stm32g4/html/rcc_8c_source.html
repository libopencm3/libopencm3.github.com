362void rcc_osc_ready_int_clear(enum rcc_osc osc)
363{
364 switch (osc) {
365 case RCC_HSI48:
366 RCC_CICR |= RCC_CICR_HSI48RDYC;
367 break;
368 case RCC_PLL:
369 RCC_CICR |= RCC_CICR_PLLRDYC;
370 break;
371 case RCC_HSE:
372 RCC_CICR |= RCC_CICR_HSERDYC;
373 break;
374 case RCC_HSI16:
375 RCC_CICR |= RCC_CICR_HSIRDYC;
376 break;
377 case RCC_LSE:
378 RCC_CICR |= RCC_CICR_LSERDYC;
379 break;
380 case RCC_LSI:
381 RCC_CICR |= RCC_CICR_LSIRDYC;
382 break;
383 }
384}

463bool rcc_is_osc_ready(enum rcc_osc osc)
464{
465 switch (osc) {
466 case RCC_HSI48:
467 return RCC_CRRCR & RCC_CRRCR_HSI48RDY;
468 case RCC_PLL:
469 return RCC_CR & RCC_CR_PLLRDY;
470 case RCC_HSE:
471 return RCC_CR & RCC_CR_HSERDY;
472 case RCC_HSI16:
473 return RCC_CR & RCC_CR_HSIRDY;
474 case RCC_LSE:
475 return RCC_BDCR & RCC_BDCR_LSERDY;
476 case RCC_LSI:
477 return RCC_CSR & RCC_CSR_LSIRDY;
478 }
479 return false;
480}

485}

487void rcc_wait_for_sysclk_status(enum rcc_osc osc)
488{
489 switch (osc) {
490 case RCC_PLL:
491 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
492 RCC_CFGR_SWx_PLL);
493 break;
494 case RCC_HSE:
495 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
496 RCC_CFGR_SWx_HSE);
497 break;
498 case RCC_HSI16:
499 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
500 RCC_CFGR_SWx_HSI16);
501 break;
502 default:
503 /* Shouldn't be reached. */
504 break;
505 }
506}

611/**
612 * Reconfigures the main PLL for a HSE source.
613 * Any reserved bits are kept at their reset values.
614 * @param pllsrc Source for the main PLL input clock
615 * @param pllm Divider for the main PLL input clock
616 * @param plln Main PLL multiplication factor for VCO
617 * @param pllp Main PLL divider for ADC
618 * @param pllq Main PLL divider for QUADSPI, FDCAN, USB, SAI & I2S
619 * @param pllr Main PLL divider for main system clock
620 */
621void rcc_set_main_pll(uint32_t pllsrc, uint32_t pllm, uint32_t plln,
622 uint32_t pllp, uint32_t pllq, uint32_t pllr)
623{
624 bool pllpen = (pllp != 0);
625 bool pllqen = (pllq != 0);
626 bool pllren = (pllr != 0);
627
628 pllm -= 1;
629
630 uint32_t pllpdiv = pllp;
631 pllp = (pllpdiv == 17);
632 if ((pllpdiv == 7) || (pllpdiv == 17)) {
633 pllpdiv = 0;
634 }
635
636 pllr = (pllr >> 1) - 1;
637 pllq = (pllq >> 1) - 1;
638
639 RCC_PLLCFGR = ((pllsrc & RCC_PLLCFGR_PLLSRC_MASK) << RCC_PLLCFGR_PLLSRC_SHIFT) |
640 ((pllm & RCC_PLLCFGR_PLLM_MASK) << RCC_PLLCFGR_PLLM_SHIFT) |
641 ((plln & RCC_PLLCFGR_PLLN_MASK) << RCC_PLLCFGR_PLLN_SHIFT) |
642 (pllpen ? RCC_PLLCFGR_PLLPEN : 0 ) |
643 (pllp ? RCC_PLLCFGR_PLLP_DIV17 : RCC_PLLCFGR_PLLP_DIV7) |

657/**
658 * Setup clocks to run from PLL.
659 *
660 * The arguments provide the pll source, multipliers, dividers, all that's
661 * needed to establish a system clock.
662 *
663 * @param clock clock information structure.
664 */
665void rcc_clock_setup_pll(const struct rcc_clock_scale *clock)
666{
667 /* Enable internal high-speed oscillator (HSI16). */
668 rcc_osc_on(RCC_HSI16);
669 rcc_wait_for_osc_ready(RCC_HSI16);
670
671 /* Select HSI16 as SYSCLK source. */
672 rcc_set_sysclk_source(RCC_CFGR_SWx_HSI16);
673
674 /* Enable external high-speed oscillator (HSE). */
675 if (clock->pll_source == RCC_PLLCFGR_PLLSRC_HSE) {
676 rcc_osc_on(RCC_HSE);
677 rcc_wait_for_osc_ready(RCC_HSE);
678 }
679
680 /* Set the VOS scale mode */
681 rcc_periph_clock_enable(RCC_PWR);
682 pwr_set_vos_scale(clock->vos_scale);
683
684 if (clock->boost) {
685 pwr_enable_boost();
686 } else {
687 pwr_disable_boost();
688 }
689
690 /*
691 * Set prescalers for AHB, ADC, APB1, APB2.
692 * Do this before touching the PLL (TODO: why?).
693 */
694 rcc_set_hpre(clock->hpre);
695 rcc_set_ppre1(clock->ppre1);
696 rcc_set_ppre2(clock->ppre2);
697
698 /* Disable PLL oscillator before changing its configuration. */
699 rcc_osc_off(RCC_PLL);
700
701 /* Configure the PLL oscillator. */
702 rcc_set_main_pll(clock->pll_source,
703 clock->pllm, clock->plln,
704 clock->pllp, clock->pllq, clock->pllr);
705
706 /* Enable PLL oscillator and wait for it to stabilize. */
707 rcc_osc_on(RCC_PLL);
708 rcc_wait_for_osc_ready(RCC_PLL);
709
710 /* Configure flash settings. */
711 if (clock->flash_config & FLASH_ACR_DCEN) {
712 flash_dcache_enable();
713 } else {
714 flash_dcache_disable();
715 }
716 if (clock->flash_config & FLASH_ACR_ICEN) {
717 flash_icache_enable();
718 } else {
719 flash_icache_disable();
720 }
721 flash_set_ws(clock->flash_waitstates);
722
723 /* Select PLL as SYSCLK source. */
724 rcc_set_sysclk_source(RCC_CFGR_SWx_PLL);
725
726 /* Wait for PLL clock to be selected. */
727 rcc_wait_for_sysclk_status(RCC_PLL);
728
729 /* Set the peripheral clock frequencies used. */
730 rcc_ahb_frequency = clock->ahb_frequency;
731 rcc_apb1_frequency = clock->apb1_frequency;
732 rcc_apb2_frequency = clock->apb2_frequency;
733
734 /* Disable internal high-speed oscillator. */
735 if (clock->pll_source == RCC_PLLCFGR_PLLSRC_HSE) {
736 rcc_osc_off(RCC_HSI16);
737 }
738}

753 *
754 * The 48 MHz clock is derived from one of the four following sources:
755 * - PLLQ VCO (RCC_CCIPR_CLK48_PLLQ)
756 * - HSI48 internal oscillator (RCC_CCIPR_CLK48_HSI48)

766static uint32_t rcc_get_clksel_freq(uint8_t shift) {
767 uint8_t clksel = (RCC_CCIPR >> shift) & RCC_CCIPR_SEL_MASK;
768 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
769 switch (clksel) {
770 case RCC_CCIPR_USARTxSEL_PCLK:
771 return rcc_apb1_frequency;
772 case RCC_CCIPR_USARTxSEL_SYSCLK:
773 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
774 case RCC_CCIPR_USARTxSEL_LSE:
775 return 32768;
776 case RCC_CCIPR_USARTxSEL_HSI16:
777 return 16000000U;
778 }
779 cm3_assert_not_reached();
780}
