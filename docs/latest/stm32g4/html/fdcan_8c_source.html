46 /** Routine implementing FDCAN_CCCR's INIT bit manipulation.
47  *
48  * This routine will change INIT bit and wait for it to actually
49  * change its value. If change won't happen before timeout,
50  * error is signalized. If INIT bit already has value which
51  * should be set, this function will return immediately.
52  *
53  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
54  * @param [in] set new value of INIT, true means set
55  * @param [in] timeout Amount of busyloop cycles, function will wait for FDCAN
56  * to switch it's state. If set to 0, then function returns immediately.
57  * @returns FDCAN_E_OK on success, FDCAN_E_TIMEOUT if INIT bit value
58  * didn't change before timeout has expired.
59  */
60 static int fdcan_cccr_init_cfg(uint32_t canport, bool set, uint32_t timeout)
61 {
62  uint32_t expected;
63  uint32_t wait_ack;
64 
65  if (set) {
66  if ((FDCAN_CCCR(canport) & FDCAN_CCCR_INIT) == FDCAN_CCCR_INIT) {
67  /* Already there, sir */
68  return FDCAN_E_OK;
69  }
70 
71  FDCAN_CCCR(canport) |= FDCAN_CCCR_INIT;
72  expected = FDCAN_CCCR_INIT;
73  } else {
74  if ((FDCAN_CCCR(canport) & FDCAN_CCCR_INIT) == 0) {
75  /* Already there, sir */
76  return FDCAN_E_OK;
77  }
78 
79  FDCAN_CCCR(canport) &= ~FDCAN_CCCR_INIT;
80  expected = 0;
81  }
82 
83  /* Wait, until INIT bit is acknowledged */
84  wait_ack = timeout;
85  while ((wait_ack--) &&
86  ((FDCAN_CCCR(canport) & FDCAN_CCCR_INIT) == expected));
87 
88  if ((FDCAN_CCCR(canport) & FDCAN_CCCR_INIT) == expected) {
89  return FDCAN_E_OK;
90  } else {
91  return FDCAN_E_TIMEOUT;
92  }
93 }

95 /** Return ID of next free Tx buffer.
96  *
97  * Examines transmit buffer allocation in message RAM
98  * and returns ID of buffer, which is free.
99  *
100  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
101  * @returns Non-negative number ID of Tx buffer which is free,
102  * or FDCAN_E_BUSY if no Tx buffer is available
103  */
104 static int fdcan_get_free_txbuf(uint32_t canport)
105 {
106  if ((FDCAN_TXBRP(canport) & FDCAN_TXBRP_TRP0) == 0) {
107  return 0;
108  } else if ((FDCAN_TXBRP(canport) & FDCAN_TXBRP_TRP1) == 0) {
109  return 1;
110  } else if ((FDCAN_TXBRP(canport) & FDCAN_TXBRP_TRP2) == 0) {
111  return 2;
112  }
113 
114  return FDCAN_E_BUSY;
115 }

117 /** Returns fill state and next available get index from receive FIFO.
118  *
119  * Examines FDCAN receive FIFO and returns fill status of FIFO and ID of
120  * next message available for reading. If fill status is 0 (FIFO is empty),
121  * then get index is undefined.
122  *
123  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
124  * @param [in] fifo_id ID of fifo queried (currently 0 or 1)
125  * @param [out] get_index Address of buffer where next get index will be stored
126  * @param [out] pending_frames Address of buffer where amount of pending frame will be stored.
127  */
128 static void fdcan_get_fill_rxfifo(uint32_t canport, uint8_t fifo_id, unsigned *get_index,
129  unsigned *pending_frames)
130 {
131  *get_index = (FDCAN_RXFIS(canport, fifo_id) >> FDCAN_RXFIFO_GI_SHIFT)
132  & FDCAN_RXFIFO_GI_MASK;
133 
134  *pending_frames = (FDCAN_RXFIS(canport, fifo_id) >> FDCAN_RXFIFO_FL_SHIFT)
135  & FDCAN_RXFIFO_FL_MASK;
136 }

138 /** Obtain address of FDCAN Message RAM for certain FDCAN block.
139  *
140  * @param [in] canport identification of FDCAN block. See @ref fdcan_block.
141  * @return Address of Message RAM for given FDCAN block or null pointer
142  * if FDCAN block identification is invalid.
143  */
144 static struct fdcan_message_ram *fdcan_get_msgram_addr(uint32_t canport)
145 {
146  /* This piece of code may look wrong, after one examines
147  * STM32G4 datasheet and/or g4/memorymap.h. There are three
148  * memory regions defined for FDCANx_RAM_BASE. They are 0x400
149  * bytes apart as per chapter 2.2.2 of [RM0440].
150  *
151  * It turns out, that these addresses are not in line with what
152  * is specified later in chapter 44.3.3 of [RM0440]. There it is
153  * stated, that message RAMs are packed and in case of multiple
154  * FDCAN blocks, message RAM for n-th FDCAN starts at address
155  * end of (n-1)-th block + 4 (explicitly, offset 0x354).
156  *
157  * It turns out, that this statement is also false! In fact FDCAN
158  * message RAMs are packed tightly and n-th block starts immediately
159  * after (n-1)-th block ends. Thus offset is going to be computed
160  * using formula:
161  *
162  * FDCAN1_RAM_BASE + (block_id * sizeof(struct fdcan_message_ram))
163  */
164  if (canport == CAN1) {
165  return (struct fdcan_message_ram *) (FDCAN1_RAM_BASE + 0);
166  } else if (canport == CAN2) {
167  return (struct fdcan_message_ram *)
168  (FDCAN1_RAM_BASE + sizeof(struct fdcan_message_ram));
169  } else if (canport == CAN3) {
170  return (struct fdcan_message_ram *)
171  (FDCAN1_RAM_BASE + (2 * sizeof(struct fdcan_message_ram)));
172  }
173 
174  return NULL;
175 }

177 /** Converts frame length to DLC value.
178  *
179  * Works for both CAN and FDCAN frame lengths. If length
180  * is invalid value, then returns 0xFF.
181  *
182  * @param [in] length intended frame payload length in bytes
183  * @returns DLC value representing lengths or 0xFF if length cannot
184  * be encoded into DLC format (applies only to FDCAN frame lengths)
185  */
186 static uint32_t fdcan_length_to_dlc(uint8_t length)
187 {
188  if (length <= 8) {
189  return length;
190  } else if (length <= 24) {
191  if ((length % 4) != 0) {
192  return 0xFF;
193  }
194  return 8 + ((length - 8) / 4);
195  } else {
196  if ((length % 16) != 0) {
197  return 0xFF;
198  }
199  return 11 + (length / 16);
200  }
201 }
202 
203 /** Converts DLC value into frame payload length.
204  *
205  * Works for both CAN and FDCAN DLC values.
206  *
207  * @param [in] dlc DLC value
208  * @returns data payload length in bytes
209  */
210 static uint8_t fdcan_dlc_to_length(uint32_t dlc)
211 {
212  if (dlc <= 8) {
213  return dlc;
214  } else if (dlc <= 12) {
215  return 8 + ((dlc - 8) * 4);
216  } else {
217  return 16 + ((dlc - 12) * 16);
218  }
219 }

227 /** Put FDCAN block into INIT mode for setup
228  *
229  * Initialize the selected CAN peripheral block. This function will switch CAN block
230  * into initialization mode. CAN block is then left in initialization mode in order to
231  * perform setup, which can't be adjusted once FDCAN block is started. It is mandatory
232  * to call at least @ref fdcan_set_can function to configure basic timing values for
233  * CAN 2.0 operation. Functions which only have effect, if FDCAN block is in INIT mode
234  * are:
235  * * @ref fdcan_set_can
236  * * @ref fdcan_set_fdcan
237  * * @ref fdcan_init_filter
238  * * @ref fdcan_set_test
239  *
240  * You can check if FDCAN block is in INIT mode or it is started using
241  * @ref fdcan_get_init_state.
242  *
243  * @param[in] canport CAN register base address. See @ref fdcan_block.
244  * @param [in] timeout Amount of empty busy loops, which routine should wait for FDCAN
245  * confirming that it entered INIT mode. If set to 0, function will return
246  * immediately.
247  * @returns Operation error status. See @ref fdcan_error.
248  */
249 int fdcan_init(uint32_t canport, uint32_t timeout)
250 {
251  if (fdcan_cccr_init_cfg(canport, true, timeout) != 0) {
252  return FDCAN_E_TIMEOUT;
253  }
254 
255  FDCAN_CCCR(canport) |= FDCAN_CCCR_CCE;
256 
257  return FDCAN_E_OK;
258 }

260 /** Set essential FDCAN block parameters for plain CAN operation
261  *
262  * Allows configuration of prescalers and essential transmit and FIFO behavior
263  * used during transmission in plain CAN 2.0 mode. In this mode FDCAN frame format
264  * is not available nor is possible to use fast bitrates.
265  * This function does neither enable FD-CAN mode after reset nor disable it
266  * after re-entering INIT mode of previously configured block. Timing values set
267  * here are valid for both arbitration phase of all frames and for data phase of
268  * both CAN and FDCAN frames, which don't use bitrate switching. This function can
269  * only be called after FDCAN block has been switched into INIT mode.
270  * It is possible to receive FDCAN frames even if FDCAN block is configured only using
271  * this function as long as bitrate switching is not used.
272  *
273  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
274  * @param [in] auto_retry_disable Disable automatic frame retransmission on error
275  * @param [in] rx_fifo_locked Enable FIFO locked mode. Upon FIFO overflow all received
276  * messages are discarded.
277  * @param [in] tx_queue_mode Enable transmission queue mode. Otherwise transmission
278  * works in FIFO mode.
279  * @param [in] silent Enable silent mode. Transmitter stays recessive all the time.
280  * @param [in] n_sjw Resynchronization time quanta jump width
281  * @param [in] n_ts1 Time segment 1 time quanta
282  * @param [in] n_ts2 Time segment 2 time quanta
283  * @param [in] n_br_presc Arbitration phase / CAN mode bitrate prescaler
284  */
285 void fdcan_set_can(uint32_t canport, bool auto_retry_disable, bool rx_fifo_locked,
286  bool tx_queue_mode, bool silent, uint32_t n_sjw, uint32_t n_ts1, uint32_t n_ts2,
287  uint32_t n_br_presc)
288 {
289  FDCAN_NBTP(canport) = (n_sjw << FDCAN_NBTP_NSJW_SHIFT)
290  | (n_ts1 << FDCAN_NBTP_NTSEG1_SHIFT)
291  | (n_ts2 << FDCAN_NBTP_NTSEG2_SHIFT)
292  | (n_br_presc << FDCAN_NBTP_NBRP_SHIFT);
293 
294  if (tx_queue_mode) {
295  FDCAN_TXBC(canport) |= FDCAN_TXBC_TFQM;
296  } else {
297  FDCAN_TXBC(canport) &= ~FDCAN_TXBC_TFQM;
298  }
299 
300  if (auto_retry_disable) {
301  FDCAN_CCCR(canport) |= FDCAN_CCCR_DAR;
302  } else {
303  FDCAN_CCCR(canport) &= ~FDCAN_CCCR_DAR;
304  }
305 
306  if (silent) {
307  FDCAN_CCCR(canport) |= FDCAN_CCCR_MON;
308  } else {
309  FDCAN_CCCR(canport) &= ~FDCAN_CCCR_MON;
310  }
311 
312  if (rx_fifo_locked) {
313  FDCAN_RXGFC(canport) &= ~(FDCAN_RXGFC_F1OM | FDCAN_RXGFC_F0OM);
314  } else {
315  FDCAN_RXGFC(canport) |= FDCAN_RXGFC_F1OM | FDCAN_RXGFC_F0OM;
316  }
317 
318 }

320 /** Set FDCAN block parameters for FDCAN transmission
321  *
322  * Enables and configures parameters related to FDCAN transmission. This function
323  * allows configuration of bitrate switching, FDCAN frame format and fast mode
324  * timing. This function can only be called if FDCAN block is in INIT mode.
325  * It is safe to call this function on previously configured block in order
326  * to enable/disable/change FDCAN mode parameters. Non-FDCAN parameters won't
327  * be affected.
328  *
329  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
330  * @param [in] brs_enable Enable FDCAN bitrate switching for fast mode operation
331  * @param [in] fd_op_enable Enable transmission of FDCAN-formatted frames
332  * @param [in] f_sjw Resynchronization time quanta jump width in fast mode
333  * @param [in] f_ts1 Time segment 1 time quanta in fast mode
334  * @param [in] f_ts2 Time segment 2 time quanta in fast mode
335  * @param [in] f_br_presc Fast mode operation bitrate prescaler
336  */
337 void fdcan_set_fdcan(uint32_t canport, bool brs_enable, bool fd_op_enable,
338  uint32_t f_sjw, uint32_t f_ts1, uint32_t f_ts2, uint32_t f_br_presc)
339 {
340  FDCAN_DBTP(canport) = (f_sjw << FDCAN_DBTP_DSJW_SHIFT)
341  | (f_ts1 << FDCAN_DBTP_DTSEG1_SHIFT)
342  | (f_ts2 << FDCAN_DBTP_DTSEG2_SHIFT)
343  | (f_br_presc << FDCAN_DBTP_DBRP_SHIFT);
344 
345  if (fd_op_enable) {
346  FDCAN_CCCR(canport) |= FDCAN_CCCR_FDOE;
347  } else {
348  FDCAN_CCCR(canport) &= ~FDCAN_CCCR_FDOE;
349  }
350 
351  if (brs_enable) {
352  FDCAN_CCCR(canport) |= FDCAN_CCCR_BRSE;
353  } else {
354  FDCAN_CCCR(canport) &= ~FDCAN_CCCR_BRSE;
355  }
356 }

425 /** Configure amount of filters and initialize filtering block.
426  *
427  * This function allows to configure global amount of filters present.
428  * FDCAN block will only ever check as many filters as this function configures.
429  * Function will also clear all filter blocks to zero values. This function
430  * can be only called after @ref fdcan_init has already been called and
431  * @ref fdcan_start has not been called yet as registers holding filter
432  * count are write-protected unless FDCAN block is in INIT mode. It is possible
433  * to reconfigure filters (@ref fdcan_set_std_filter and @ref fdcan_set_ext_filter)
434  * after FDCAN block has already been started.
435  *
436  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
437  * @param [in] std_filt requested amount of standard ID filter rules (0-28)
438  * @param [in] ext_filt requested amount of extended ID filter rules (0-8)
439  */
440 void fdcan_init_filter(uint32_t canport, uint8_t std_filt, uint8_t ext_filt)
441 {
442  struct fdcan_message_ram *ram = fdcan_get_msgram_addr(canport);
443 
444  /* Only perform initialization of message RAM if there are
445  * any filters required
446  */
447  if (std_filt > 0) {
448  FDCAN_RXGFC(canport) =
449  (FDCAN_RXGFC(canport) & ~(FDCAN_RXGFC_LSS_MASK << FDCAN_RXGFC_LSS_SHIFT))
450  | (std_filt << FDCAN_RXGFC_LSS_SHIFT);
451 
452 
453  for (int q = 0; q < FDCAN_SFT_MAX_NR; ++q) {
454  ram->lfssa[q].type_id1_conf_id2 = 0;
455  }
456  } else {
457  /* Reset filter count to zero */
458  FDCAN_RXGFC(canport) =
459  (FDCAN_RXGFC(canport) & ~(FDCAN_RXGFC_LSS_MASK << FDCAN_RXGFC_LSS_SHIFT));
460  }
461 
462  if (ext_filt > 0) {
463  FDCAN_RXGFC(canport) =
464  (FDCAN_RXGFC(canport) & ~(FDCAN_RXGFC_LSE_MASK << FDCAN_RXGFC_LSE_SHIFT))
465  | (ext_filt << FDCAN_RXGFC_LSE_SHIFT);
466 
467  for (int q = 0; q < FDCAN_EFT_MAX_NR; ++q) {
468  ram->lfesa[q].conf_id1 = 0;
469  ram->lfesa[q].type_id2 = 0;
470  }
471  } else {
472  /* Reset filter count to zero */
473  FDCAN_RXGFC(canport) =
474  (FDCAN_RXGFC(canport) & ~(FDCAN_RXGFC_LSE_MASK << FDCAN_RXGFC_LSE_SHIFT));
475  }
476 }

478 /** Configure filter rule for standard ID frames.
479  *
480  * Sets up filter rule for frames having standard ID. Each FDCAN block can
481  * have its own set of filtering rules. It is only possible to configure as
482  * many filters as was configured previously using @ref fdcan_init_filter.
483  *
484  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
485  * @param [in] nr number of filter to be configured
486  * @param [in] id_list_mode Mode in which id1 and id2 are used to match the rule.
487  * See @ref fdcan_sft.
488  * @param [in] id1 standard ID for matching. Used as exact value, lower bound or bit
489  * pattern depending on matching mode selected
490  * @param [in] id2 standard ID or bitmask. Used as exact value, upper bound or bit mask
491  * depending on matching mode selected
492  * @param [in] action Action performed if filtering rule matches frame ID.
493  * See @ref fdcan_sfec.
494  */
495 void fdcan_set_std_filter(uint32_t canport, uint32_t nr,
496  uint8_t id_list_mode, uint32_t id1, uint32_t id2,
497  uint8_t action)
498 {
499  struct fdcan_message_ram *ram = fdcan_get_msgram_addr(canport);
500 
501  /* id_list_mode and action are passed unguarded. Simply use
502  * defines and it will be OK. id1 and id2 are masked for
503  * correct size, because it is way too simple to pass ID
504  * larger than 11 bits unintentionally. It then leads to all
505  * kind of extremely weird errors while excessive ID bits
506  * overflow into flags. This tends to be extremely time
507  * consuming to debug.
508  */
509  ram->lfssa[nr].type_id1_conf_id2 =
510  (id_list_mode << FDCAN_SFT_SHIFT)
511  | (action << FDCAN_SFEC_SHIFT)
512  | ((id1 & FDCAN_SFID1_MASK) << FDCAN_SFID1_SHIFT)
513  | ((id2 & FDCAN_SFID2_MASK) << FDCAN_SFID2_SHIFT);
514 
515  return;
516 }

519  *
520  * Sets up filter rule for frames having extended ID. Each FDCAN block can
521  * have its own set of filtering rules. It is only possible to configure as
522  * many filters as was configured previously using @ref fdcan_init_filter.
523  *
524  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
525  * @param [in] nr number of filter to be configured
526  * @param [in] id_list_mode mode in which id1 and id2 are used to match the rule.
527  * See @ref fdcan_eft.
528  * @param [in] id1 extended ID for matching. Used as exact value, lower bound or bit
529  * pattern depending on matching mode selected
530  * @param [in] id2 extended ID or bitmask. Used as exact value, upper bound or bit mask
531  * depending on matching mode selected
532  * @param [in] action Action performed if filtering rule matches frame ID.
533  * See @ref fdcan_efec.

535 void fdcan_set_ext_filter(uint32_t canport, uint32_t nr,
536  uint8_t id_list_mode, uint32_t id1, uint32_t id2,
537  uint8_t action)
538 {

550 /** Transmit Message using FDCAN
551  *
552  * @param [in] canport CAN block register base. See @ref fdcan_block.
553  * @param [in] id Message ID
554  * @param [in] ext Extended message ID
555  * @param [in] rtr Request transmit
556  * @param [in] fdcan_fmt Use FDCAN format
557  * @param [in] btr_switch Switch bitrate for data portion of frame
558  * @param [in] length Message payload length. Must be valid CAN or FDCAN frame length
559  * @param [in] data Message payload data
560  * @returns int 0, 1 or 2 on success and depending on which outgoing mailbox got
561  * selected. Otherwise returns error code. For error codes, see @ref fdcan_error.
562  */
563 int fdcan_transmit(uint32_t canport, uint32_t id, bool ext, bool rtr,
564  bool fdcan_fmt, bool btr_switch, uint8_t length, const uint8_t *data)
565 {
566  int mailbox;
567  uint32_t dlc, flags = 0;
568 
569  mailbox = fdcan_get_free_txbuf(canport);
570 
571  if (mailbox == FDCAN_E_BUSY) {
572  return mailbox;
573  }
574 
575  struct fdcan_message_ram *ram = fdcan_get_msgram_addr(canport);
576 
577  /* Early check: if FDCAN message lentgh is > 8, it must be
578  * a multiple of 4 *and* fdcan format must be enabled.
579  */
580  dlc = fdcan_length_to_dlc(length);
581 
582  if (dlc == 0xFF) {
583  return FDCAN_E_INVALID;
584  }
585 
586  if (ext) {
587  ram->tx_buffer[mailbox].identifier_flags = FDCAN_FIFO_XTD
588  | ((id & FDCAN_FIFO_EID_MASK) << FDCAN_FIFO_EID_SHIFT);
589  } else {
590  ram->tx_buffer[mailbox].identifier_flags =
591  (id & FDCAN_FIFO_SID_MASK) << FDCAN_FIFO_SID_SHIFT;
592  }
593 
594  if (rtr) {
595  ram->tx_buffer[mailbox].identifier_flags |= FDCAN_FIFO_RTR;
596  }
597 
598  if (fdcan_fmt) {
599  flags |= FDCAN_FIFO_FDF;
600  }
601 
602  if (btr_switch) {
603  flags |= FDCAN_FIFO_BRS;
604  }
605 
606  ram->tx_buffer[mailbox].evt_fmt_dlc_res =
607  (dlc << FDCAN_FIFO_DLC_SHIFT) | flags;
608 
609  for (int q = 0; q < length; q += 4) {
610  ram->tx_buffer[mailbox].data[q / 4] = *((uint32_t *) &data[q]);
611  }
612 
613  FDCAN_TXBAR(canport) |= 1 << mailbox;
614 
615  return mailbox;
616 }

618 /** Receive Message from FDCAN FIFO
619  *
620  * Reads one message from receive FIFO. Returns message ID, type of ID, message length
621  * and message payload. It is mandatory to provide valid pointers to suitably sized buffers
622  * for these outputs. Additionally, it is optinally possible to provide non-zero pointer to
623  * obtain filter identification, request of transmission flag and message timestamp.
624  * If pointers provided for optional outputs are NULL, then no information is returned
625  * for given pointer.
626  *
627  * @param [in] canport FDCAN block base address. See @ref fdcan_block
628  * @param [in] fifo_id FIFO id.
629  * @param [in] release Release the FIFO automatically after copying data out
630  * @param [out] id Returned message ID. Mandatory.
631  * @param [out] ext Returned type of the message ID (true if extended). Mandatory.
632  * @param [out] rtr Returnes flag if request of transmission was requested. Optional.
633  * @param [out] fmi Returned ID of the filter which matched this frame. Optional.
634  * @param [out] length Length of message payload in bytes. Mandatory.
635  * @param [out] data Buffer for storage of message payload data. Mandatory.
636  * @param [out] timestamp Returned timestamp of received frame. Optional.
637  * @returns Operation error status. See @ref fdcan_error.
638  */
639 int fdcan_receive(uint32_t canport, uint8_t fifo_id, bool release, uint32_t *id,
640  bool *ext, bool *rtr, uint8_t *fmi, uint8_t *length,
641  uint8_t *data, uint16_t *timestamp)
642 {
643  const struct fdcan_message_ram *ram = fdcan_get_msgram_addr(canport);
644 
645  const struct fdcan_rx_fifo_element *fifo;
646 
647  unsigned pending_frames, get_index, dlc, len;
648 
649  fdcan_get_fill_rxfifo(canport, fifo_id, &get_index, &pending_frames);
650 
651  fifo = ram->rx_fifo[fifo_id];
652 
653  if (pending_frames == 0) {
654  return FDCAN_E_NOTAVAIL;
655  }
656 
657  dlc = (fifo[get_index].filt_fmt_dlc_ts >> FDCAN_FIFO_DLC_SHIFT)
658  & FDCAN_FIFO_DLC_MASK;
659 
660  len = fdcan_dlc_to_length(dlc);
661 
662  *length = len;
663  if ((fifo[get_index].identifier_flags & FDCAN_FIFO_XTD) == FDCAN_FIFO_XTD) {
664  *ext = true;
665  *id = (fifo[get_index].identifier_flags >> FDCAN_FIFO_EID_SHIFT)
666  & FDCAN_FIFO_EID_MASK;
667  } else {
668  *ext = false;
669  *id = (fifo[get_index].identifier_flags >> FDCAN_FIFO_SID_SHIFT)
670  & FDCAN_FIFO_SID_MASK;
671  }
672 
673  if (timestamp) {
674  *timestamp = (uint16_t) (fifo[get_index].filt_fmt_dlc_ts >> FDCAN_FIFO_RXTS_SHIFT)
675  & FDCAN_FIFO_RXTS_MASK;
676  }
677 
678  if (fmi) {
679  *fmi = (uint8_t) (fifo[get_index].filt_fmt_dlc_ts >> FDCAN_FIFO_MM_SHIFT)
680  & FDCAN_FIFO_MM_MASK;
681  }
682 
683  if (rtr) {
684  *rtr = ((fifo[get_index].identifier_flags & FDCAN_FIFO_RTR) == FDCAN_FIFO_RTR);
685  }
686 
687  for (unsigned int q = 0; q < len; q += 4) {
688  *((uint32_t *) &data[q]) = fifo[get_index].data[q / 4];
689  }
690 
691  if (release) {
692  FDCAN_RXFIA(canport, fifo_id) |= get_index << FDCAN_RXFIFO_AI_SHIFT;
693  }
694 
695  return FDCAN_E_OK;
696 }

698 /** Release receive oldest FIFO entry.
699  *
700  * This function will mask oldest entry in FIFO as released making
701  * space for another received frame. This function can be used if
702  * fdcan_receive was called using release = false. If used in other
703  * case, then messages can get lost.
704  *
705  * @param [in] canport FDCAN block base address. See @ref fdcan_block.
706  * @param [in] fifo_id ID of FIFO where release should be performed (0 or 1)
707  */
708 void fdcan_release_fifo(uint32_t canport, uint8_t fifo_id)
709 {
710  unsigned pending_frames, get_index;
711 
712  get_index = (FDCAN_RXFIS(canport, fifo_id) >> FDCAN_RXFIFO_GI_SHIFT)
713  & FDCAN_RXFIFO_GI_SHIFT;
714 
715  pending_frames = (FDCAN_RXFIS(canport, fifo_id) >> FDCAN_RXFIFO_FL_SHIFT)
716  & FDCAN_RXFIFO_FL_SHIFT;
717 
718  if (pending_frames) {
719  FDCAN_RXFIA(canport, fifo_id) |= get_index << FDCAN_RXFIFO_AI_SHIFT;
720  }
721 }
