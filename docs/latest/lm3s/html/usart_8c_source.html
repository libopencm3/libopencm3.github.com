33void usart_send_blocking(uint32_t usart, uint16_t data)

39bool usart_is_recv_ready(uint32_t usart)
40{
41 return ((USART_FR(usart) & USART_FR_RXFE) == 0);

44bool usart_is_send_ready(uint32_t usart)
