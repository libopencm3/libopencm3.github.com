53/** Initialize the USB device controller hardware of the EFM32HG. */
54static usbd_device *efm32hg_usbd_init(void)
55{
56 /* Enable peripheral clocks required for USB */
57 cmu_periph_clock_enable(CMU_USB);
58 cmu_periph_clock_enable(CMU_USBC);
59 cmu_periph_clock_enable(CMU_LE);
60
61 /* Select LFRCO as LFCCLK clock */
62 CMU_LFCLKSEL = CMU_LFCLKSEL_LFC_LFRCO;
63
64 /* Enable the USBLE peripheral clock (sits on LFCCLK) */
65 cmu_periph_clock_enable(CMU_USBLE);
66
67 /* Calibrate USB based on communications */
68 CMU_USHFRCOCONF = CMU_USHFRCOCONF_BAND_48MHZ;
69
70 /* Enable USHFRCO Clock Recovery mode. */
71 CMU_USBCRCTRL |= CMU_USBCRCTRL_EN;
72
73 /* Select USHFRCO as clock source for USB */
74 cmu_osc_on(USHFRCO);
75 cmu_wait_for_osc_ready(USHFRCO);
76
77 /* Set up the USB clock source */
78 cmu_set_usbclk_source(USHFRCO);
79 cmu_wait_for_usbclk_selected(USHFRCO);
80
81 /* Turn off all Low Energy Mode (LEM) features. */
82 USB_CTRL = 0;
83
84 /* Initialize USB core */
85 USB_ROUTE = USB_ROUTE_PHYPEN; /* Enable PHY pins. */
86
87 /* Wait for AHB idle. */
88 while (!(OTG_FS_GRSTCTL & OTG_GRSTCTL_AHBIDL));
89 /* Do core soft reset. */
90 OTG_FS_GRSTCTL |= OTG_GRSTCTL_CSRST;
91 while (OTG_FS_GRSTCTL & OTG_GRSTCTL_CSRST);
92
93 /* Explicitly enable DP pullup (not all cores do this by default) */
94 OTG_FS_DCTL &= ~OTG_DCTL_SDIS;
95
96 /* Force peripheral only mode. */
97 OTG_FS_GUSBCFG |= OTG_GUSBCFG_FDMOD | OTG_GUSBCFG_TRDT_MASK;
98
99 OTG_FS_GINTSTS = OTG_GINTSTS_MMIS;
100
101 /* Full speed device. */
102 OTG_FS_DCFG |= OTG_DCFG_DSPD;
103
104 /* Restart the PHY clock. */
105 OTG_FS_PCGCCTL = 0;
106
107 OTG_FS_GRXFSIZ = efm32hg_usb_driver.rx_fifo_size;
108 _usbd_dev.fifo_mem_top = efm32hg_usb_driver.rx_fifo_size;
109
110 /* Unmask interrupts for TX and RX. */
111 OTG_FS_GAHBCFG |= OTG_GAHBCFG_GINT;
112 OTG_FS_GINTMSK = OTG_GINTMSK_ENUMDNEM |
113 OTG_GINTMSK_RXFLVLM |
114 OTG_GINTMSK_IEPINT |
115 OTG_GINTMSK_USBSUSPM |
116 OTG_GINTMSK_WUIM;
117 OTG_FS_DAINTMSK = 0xF;
118 OTG_FS_DIEPMSK = OTG_DIEPMSK_XFRCM;
119
120 return &_usbd_dev;
121}
