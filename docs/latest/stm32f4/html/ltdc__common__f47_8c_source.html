70 void ltdc_setup_windowing(uint8_t layer_number,

73 {
74  active_width += h_back_porch - 1;

76  /*assert((h_back_porch & 0xfff == h_back_porch) &&

79  (active_height & 0xfff == active_height));*/
80  LTDC_LxWHPCR(layer_number) = (active_width << 16) |
