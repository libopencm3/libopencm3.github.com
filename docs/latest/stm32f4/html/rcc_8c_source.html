460 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
461 {
462  switch (osc) {
463  case RCC_PLL:
464  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
465  RCC_CFGR_SWS_PLL);
466  break;
467  case RCC_HSE:
468  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
469  RCC_CFGR_SWS_HSE);
470  break;
471  case RCC_HSI:
472  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
473  RCC_CFGR_SWS_HSI);
474  break;
475  default:
476  /* Shouldn't be reached. */
477  break;
478  }
479 }

547  * @param n valid range depends on target device, check your RefManual.

550 void rcc_plli2s_config(uint16_t n, uint8_t r)

557 /**
558  * Set the dividers for the PLLSAI clock outputs
559  * divider p is only available on F4x9 parts, pass 0 for other parts.
560  * @param n valid range is 49..432
561  * @param p 0 if unused, @ref rcc_pllsaicfgr_pllsaip
562  * @param q valid range is 2..15
563  * @param r valid range is 2..7
564  * @sa rcc_pllsai_postscalers
565  */
566 void rcc_pllsai_config(uint16_t n, uint16_t p, uint16_t q, uint16_t r)
567 {
568  RCC_PLLSAICFGR = (
569  ((n & RCC_PLLSAICFGR_PLLSAIN_MASK) << RCC_PLLSAICFGR_PLLSAIN_SHIFT) |
570  ((p & RCC_PLLSAICFGR_PLLSAIP_MASK) << RCC_PLLSAICFGR_PLLSAIP_SHIFT) |
571  ((q & RCC_PLLSAICFGR_PLLSAIQ_MASK) << RCC_PLLSAICFGR_PLLSAIQ_SHIFT) |
572  ((r & RCC_PLLSAICFGR_PLLSAIR_MASK) << RCC_PLLSAICFGR_PLLSAIR_SHIFT));
573 }

576 /**
577  * Set the dedicated dividers after the PLLSAI configuration.
578  *
579  * @param q dedicated PLLSAI divider, for either A or B
580  * @param r dedicated LCD-TFT divider, see LTDC
581  * @sa rcc_pllsai_config
582  */
583 void rcc_pllsai_postscalers(uint8_t q, uint8_t r)
584 {
585  uint32_t reg32 = RCC_DCKCFGR;
586  reg32 &= ((RCC_DCKCFGR_PLLSAIDIVR_MASK << RCC_DCKCFGR_PLLSAIDIVR_SHIFT)
587  | (RCC_DCKCFGR_PLLSAIDIVQ_MASK << RCC_DCKCFGR_PLLSAIDIVQ_SHIFT));
588  RCC_DCKCFGR = reg32 | ((q << RCC_DCKCFGR_PLLSAIDIVQ_SHIFT) |
589  (r << RCC_DCKCFGR_PLLSAIDIVR_SHIFT));
590 }

593 void rcc_set_sysclk_source(uint32_t clk)
594 {
595  uint32_t reg32;
596 
597  reg32 = RCC_CFGR;
598  reg32 &= ~((1 << 1) | (1 << 0));
599  RCC_CFGR = (reg32 | clk);

701 /**
702  * Setup clocks to run from PLL.
703  *
704  * The arguments provide the pll source, multipliers, dividers, all that's
705  * needed to establish a system clock.
706  *
707  * @param clock clock information structure.
708  */
709 void rcc_clock_setup_pll(const struct rcc_clock_scale *clock)
710 {
711  /* Enable internal high-speed oscillator (HSI). */
712  rcc_osc_on(RCC_HSI);
713  rcc_wait_for_osc_ready(RCC_HSI);
714 
715  /* Select HSI as SYSCLK source. */
716  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
717 
718  /* Enable external high-speed oscillator (HSE). */
719  if (clock->pll_source == RCC_CFGR_PLLSRC_HSE_CLK) {
720  rcc_osc_on(RCC_HSE);
721  rcc_wait_for_osc_ready(RCC_HSE);
722  }
723 
724  /* Set the VOS scale mode */
725  rcc_periph_clock_enable(RCC_PWR);
726  pwr_set_vos_scale(clock->voltage_scale);
727 
728  /*
729  * Set prescalers for AHB, ADC, APB1, APB2.
730  * Do this before touching the PLL (TODO: why?).
731  */
732  rcc_set_hpre(clock->hpre);
733  rcc_set_ppre1(clock->ppre1);
734  rcc_set_ppre2(clock->ppre2);
735 
736  /* Disable PLL oscillator before changing its configuration. */
737  rcc_osc_off(RCC_PLL);
738 
739  /* Configure the PLL oscillator. */
740  if (clock->pll_source == RCC_CFGR_PLLSRC_HSE_CLK) {
741  rcc_set_main_pll_hse(clock->pllm, clock->plln,
742  clock->pllp, clock->pllq, clock->pllr);
743  } else {
744  rcc_set_main_pll_hsi(clock->pllm, clock->plln,
745  clock->pllp, clock->pllq, clock->pllr);
746  }
747 
748  /* Enable PLL oscillator and wait for it to stabilize. */
749  rcc_osc_on(RCC_PLL);
750  rcc_wait_for_osc_ready(RCC_PLL);
751 
752  /* Configure flash settings. */
753  if (clock->flash_config & FLASH_ACR_DCEN) {
754  flash_dcache_enable();
755  } else {
756  flash_dcache_disable();
757  }
758  if (clock->flash_config & FLASH_ACR_ICEN) {
759  flash_icache_enable();
760  } else {
761  flash_icache_disable();
762  }
763  flash_set_ws(clock->flash_config);
764 
765  /* Select PLL as SYSCLK source. */
766  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
767 
768  /* Wait for PLL clock to be selected. */
769  rcc_wait_for_sysclk_status(RCC_PLL);
770 
771  /* Set the peripheral clock frequencies used. */
772  rcc_ahb_frequency = clock->ahb_frequency;
773  rcc_apb1_frequency = clock->apb1_frequency;
774  rcc_apb2_frequency = clock->apb2_frequency;
775 
776  /* Disable internal high-speed oscillator. */
777  if (clock->pll_source == RCC_CFGR_PLLSRC_HSE_CLK) {
778  rcc_osc_off(RCC_HSI);
779  }
780 }
