49#define DMA2D_CR_MODE_M2MWB 2 /* memory to memory with blend */
50#define DMA2D_CR_MODE_R2M 3 /* register to memory */
51#define DMA2D_CR_CEIE (1 << 13)

53#define DMA2D_CR_CAEIE (1 << 11)
