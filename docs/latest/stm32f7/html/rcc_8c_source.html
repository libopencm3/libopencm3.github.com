206void rcc_wait_for_osc_ready(enum rcc_osc osc)
207{
208 switch (osc) {
209 case RCC_PLL:
210 while ((RCC_CR & RCC_CR_PLLRDY) == 0);
211 break;
212 case RCC_HSE:
213 while ((RCC_CR & RCC_CR_HSERDY) == 0);
214 break;
215 case RCC_HSI:
216 while ((RCC_CR & RCC_CR_HSIRDY) == 0);
217 break;
218 case RCC_LSE:
219 while ((RCC_BDCR & RCC_BDCR_LSERDY) == 0);
220 break;
221 case RCC_LSI:
222 while ((RCC_CSR & RCC_CSR_LSIRDY) == 0);
223 break;
224 }
225}

227void rcc_wait_for_sysclk_status(enum rcc_osc osc)
228{
229 switch (osc) {
230 case RCC_PLL:
231 while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_PLL);
232 break;
233 case RCC_HSE:
234 while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSE);
235 break;
236 case RCC_HSI:
237 while ((RCC_CFGR & ((1 << 1) | (1 << 0))) != RCC_CFGR_SWS_HSI);
238 break;
239 default:
240 /* Shouldn't be reached. */
241 break;
242 }

376void rcc_clock_setup_hse(const struct rcc_clock_scale *clock, uint32_t hse_mhz)
377{
378 uint8_t pllm = hse_mhz;
379
380 /* Enable internal high-speed oscillator. */
381 rcc_osc_on(RCC_HSI);
382 rcc_wait_for_osc_ready(RCC_HSI);
383
384 /* Select HSI as SYSCLK source. */
385 rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
386
387 /* Enable external high-speed oscillator. */
388 rcc_osc_on(RCC_HSE);
389 rcc_wait_for_osc_ready(RCC_HSE);
390
391 rcc_periph_clock_enable(RCC_PWR);
392 pwr_set_vos_scale(clock->vos_scale);
393
394 if (clock->overdrive) {
395 pwr_enable_overdrive();
396 }
397
398 /*
399 * Set prescalers for AHB, ADC, APB1, APB2.
400 * Do this before touching the PLL (TODO: why?).
401 */
402 rcc_set_hpre(clock->hpre);
403 rcc_set_ppre1(clock->ppre1);
404 rcc_set_ppre2(clock->ppre2);
405
406 /* Disable PLL oscillator before changing its configuration. */
407 rcc_osc_off(RCC_PLL);
408
409 /* Configure the PLL oscillator. */
410 rcc_set_main_pll_hse(pllm, clock->plln,
411 clock->pllp, clock->pllq);
412
413 /* Enable PLL oscillator and wait for it to stabilize. */
414 rcc_osc_on(RCC_PLL);
415 rcc_wait_for_osc_ready(RCC_PLL);
416
417 /* Configure flash settings. */
418 flash_set_ws(clock->flash_waitstates);
419 flash_art_enable();
420 flash_prefetch_enable();
421
422 /* Select PLL as SYSCLK source. */
423 rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
424
425 /* Wait for PLL clock to be selected. */
426 rcc_wait_for_sysclk_status(RCC_PLL);
427
428 /* Set the clock frequencies used. */
429 rcc_ahb_frequency = clock->ahb_frequency;
430 rcc_apb1_frequency = clock->apb1_frequency;
431 rcc_apb2_frequency = clock->apb2_frequency;
432
433 /* Disable internal high-speed oscillator. */
434 rcc_osc_off(RCC_HSI);
435}

487static uint32_t rcc_usart_i2c_clksel_freq(uint32_t apb_clk, uint8_t shift) {
488 uint8_t clksel = (RCC_DCKCFGR2 >> shift) & RCC_DCKCFGR2_UARTxSEL_MASK;
489 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
490 switch (clksel) {
491 case RCC_DCKCFGR2_UARTxSEL_PCLK:
492 return apb_clk;
493 case RCC_DCKCFGR2_UARTxSEL_SYSCLK:
494 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
495 /* This case is only valid for uarts, not for i2c! */
496 case RCC_DCKCFGR2_UARTxSEL_LSE:
497 return 32768;
498 case RCC_DCKCFGR2_UARTxSEL_HSI:
499 return 16000000U;
500 }
501 cm3_assert_not_reached();
502}

505/** @brief Get the peripheral clock speed for the USART at base specified.
506 * @param usart Base address of USART to get clock frequency for.
507 */
508uint32_t rcc_get_usart_clk_freq(uint32_t usart)
509{
510 /* F7 is highly configurable, every USART can be configured in DCKCFGR2. */
511 if (usart == USART1_BASE) {
512 return rcc_usart_i2c_clksel_freq(rcc_apb2_frequency, RCC_DCKCFGR2_UART1SEL_SHIFT);
513 } else if (usart == USART2_BASE) {
514 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART2SEL_SHIFT);
515 } else if (usart == USART3_BASE) {
516 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART3SEL_SHIFT);
517 } else if (usart == UART4_BASE) {
518 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART4SEL_SHIFT);
519 } else if (usart == UART5_BASE) {
520 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART5SEL_SHIFT);
521 } else if (usart == USART6_BASE) {
522 return rcc_usart_i2c_clksel_freq(rcc_apb2_frequency, RCC_DCKCFGR2_USART6SEL_SHIFT);
523 } else if (usart == UART7_BASE) {
524 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART7SEL_SHIFT);
525 } else { /* UART8 */
526 return rcc_usart_i2c_clksel_freq(rcc_apb1_frequency, RCC_DCKCFGR2_UART8SEL_SHIFT);
527 }
528}

531/** @brief Get the peripheral clock speed for the Timer at base specified.
532 * @param timer Base address of TIM to get clock frequency for.
533 */
534uint32_t rcc_get_timer_clk_freq(uint32_t timer)
535{
536 /* Handle APB1 timer clocks. */
537 if (timer >= TIM2_BASE && timer <= TIM14_BASE) {
538 uint8_t ppre1 = (RCC_CFGR >> RCC_CFGR_PPRE1_SHIFT) & RCC_CFGR_PPRE1_MASK;
539 return (ppre1 == RCC_CFGR_PPRE_DIV_NONE) ? rcc_apb1_frequency
540 : 2 * rcc_apb1_frequency;
541 } else {
542 uint8_t ppre2 = (RCC_CFGR >> RCC_CFGR_PPRE2_SHIFT) & RCC_CFGR_PPRE2_MASK;
543 return (ppre2 == RCC_CFGR_PPRE_DIV_NONE) ? rcc_apb2_frequency
544 : 2 * rcc_apb2_frequency;
545 }
546}
