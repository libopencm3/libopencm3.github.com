376 void rcc_clock_setup_hse(const struct rcc_clock_scale *clock, uint32_t hse_mhz)
377 {
378  uint8_t pllm = hse_mhz;
379 
380  /* Enable internal high-speed oscillator. */
381  rcc_osc_on(RCC_HSI);
382  rcc_wait_for_osc_ready(RCC_HSI);
383 
384  /* Select HSI as SYSCLK source. */
385  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
386 
387  /* Enable external high-speed oscillator. */
388  rcc_osc_on(RCC_HSE);
389  rcc_wait_for_osc_ready(RCC_HSE);
390 
391  rcc_periph_clock_enable(RCC_PWR);
392  pwr_set_vos_scale(clock->vos_scale);
393 
394  if (clock->overdrive) {
395  pwr_enable_overdrive();
396  }
397 
398  /*
399  * Set prescalers for AHB, ADC, APB1, APB2.
400  * Do this before touching the PLL (TODO: why?).
401  */
402  rcc_set_hpre(clock->hpre);
403  rcc_set_ppre1(clock->ppre1);
404  rcc_set_ppre2(clock->ppre2);
405 
406  /* Disable PLL oscillator before changing its configuration. */
407  rcc_osc_off(RCC_PLL);
408 
409  /* Configure the PLL oscillator. */
410  rcc_set_main_pll_hse(pllm, clock->plln,
411  clock->pllp, clock->pllq);
412 
413  /* Enable PLL oscillator and wait for it to stabilize. */
414  rcc_osc_on(RCC_PLL);
415  rcc_wait_for_osc_ready(RCC_PLL);
416 
417  /* Configure flash settings. */
418  flash_set_ws(clock->flash_waitstates);
419  flash_art_enable();
420  flash_prefetch_enable();
421 
422  /* Select PLL as SYSCLK source. */
423  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
424 
425  /* Wait for PLL clock to be selected. */
426  rcc_wait_for_sysclk_status(RCC_PLL);
427 
428  /* Set the clock frequencies used. */
429  rcc_ahb_frequency = clock->ahb_frequency;
430  rcc_apb1_frequency = clock->apb1_frequency;
431  rcc_apb2_frequency = clock->apb2_frequency;
432 
433  /* Disable internal high-speed oscillator. */
434  rcc_osc_off(RCC_HSI);
435 }

487 static uint32_t rcc_usart_i2c_clksel_freq(uint32_t apb_clk, uint8_t shift) {
488  uint8_t clksel = (RCC_DCKCFGR2 >> shift) & RCC_DCKCFGR2_UART1SEL_MASK;
489  uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
490  switch (clksel) {
491  case RCC_DCKCFGR2_UARTxSEL_PCLK:
492  return apb_clk;
493  case RCC_DCKCFGR2_UARTxSEL_SYSCLK:
494  return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
495  case RCC_DCKCFGR2_UARTxSEL_HSI:
496  return 16000000U;
497  default:
498  cm3_assert_not_reached();
499  }
500 }
