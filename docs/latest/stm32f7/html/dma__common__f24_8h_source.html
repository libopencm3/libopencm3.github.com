249/* --- DMA Interrupt Flag offset values ------------------------------------- */
250
251/* For API parameters. These are based on every interrupt flag and flag clear
252being at the same relative location */
253/** @defgroup dma_if_offset DMA Interrupt Flag Offsets within stream flag group.
254@ingroup dma_defines
255
256@{*/
257/** Transfer Complete Interrupt Flag */
258#define DMA_TCIF (1 << 5)
259/** Half Transfer Interrupt Flag */
260#define DMA_HTIF (1 << 4)
261/** Transfer Error Interrupt Flag */
262#define DMA_TEIF (1 << 3)
263/** Direct Mode Error Interrupt Flag */
264#define DMA_DMEIF (1 << 2)
265/** FIFO Error Interrupt Flag */
266#define DMA_FEIF (1 << 0)
267/**@}*/
268
269/* Offset within interrupt status register to start of stream interrupt flag
270 * field
271 */
272#define DMA_ISR_OFFSET(stream) (6*((stream) & 0x01)+16*(((stream) & 0x02) >> 1))
273#define DMA_ISR_FLAGS (DMA_TCIF | DMA_HTIF | DMA_TEIF | DMA_DMEIF | \
274 DMA_FEIF)
275#define DMA_ISR_MASK(stream) (DMA_ISR_FLAGS << DMA_ISR_OFFSET(stream))
