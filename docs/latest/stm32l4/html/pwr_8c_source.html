39void pwr_set_vos_scale(enum pwr_vos_scale scale)
40{
41 uint32_t reg32;
42
43 reg32 = PWR_CR1;
44 reg32 &= ~(PWR_CR1_VOS_MASK << PWR_CR1_VOS_SHIFT);
45 switch (scale) {
46 case PWR_SCALE1:
47 reg32 |= (PWR_CR1_VOS_RANGE_1 << PWR_CR1_VOS_SHIFT);
48 break;
49 case PWR_SCALE2:
50 reg32 |= (PWR_CR1_VOS_RANGE_2 << PWR_CR1_VOS_SHIFT);
51 break;
52 }
53 PWR_CR1 = reg32;
54}

56/** Disable Backup Domain Write Protection
57 *
58 * This allows backup domain registers to be changed. These registers are write
59 * protected after a reset.
60 */
61void pwr_disable_backup_domain_write_protect(void)
62{
63 PWR_CR1 |= PWR_CR1_DBP;
64}

66/** Re-enable Backup Domain Write Protection
67 *
68 * This protects backup domain registers from inadvertent change.
69 */
70void pwr_enable_backup_domain_write_protect(void)
71{
72 PWR_CR1 &= ~PWR_CR1_DBP;
73}
