38 /** @brief USART Set Baudrate.
39 
40 The baud rate is computed from the APB high-speed prescaler clock (for
41 USART1/6) or the APB low-speed prescaler clock (for other USARTs). These values
42 must be correctly set before calling this function (refer to the
43 rcc_clock_setup-* functions in RCC).
44 
45 Note: For LPUART, baudrates over 2**24 (~16.7 Mbaud) may overflow
46 the calculation and are therefore not supported by this function.
47 
48 @param[in] usart unsigned 32 bit. USART block register address base @ref
49 usart_reg_base
50 @param[in] baud unsigned 32 bit. Baud rate specified in Hz.
51 */
52 
53 void usart_set_baudrate(uint32_t usart, uint32_t baud)
54 {
55  uint32_t clock = rcc_apb1_frequency;
56 
57 #if defined USART1
58  if ((usart == USART1)
59 #if defined USART6
60  || (usart == USART6)
61 #endif
62  ) {
63  clock = rcc_apb2_frequency;
64  }
65 #endif
66 
67  /*
68  * Yes it is as simple as that. The reference manual is
69  * talking about fractional calculation but it seems to be only
70  * marketing babble to sound awesome. It is nothing else but a
71  * simple divider to generate the correct baudrate.
72  *
73  * Note: We round() the value rather than floor()ing it, for more
74  * accurate divisor selection.
75  */
76 #ifdef LPUART1
77  if (usart == LPUART1) {
78  USART_BRR(usart) = (clock / baud) * 256
79  + ((clock % baud) * 256 + baud / 2) / baud;
80  return;
81  }
82 #endif
83 
84  USART_BRR(usart) = (clock + baud / 2) / baud;
85 }

88 /** @brief USART Set Word Length.
89 
90 The word length is set to 8 or 9 bits. Note that the last bit will be a parity
91 bit if parity is enabled, in which case the data length will be 7 or 8 bits
92 respectively.
93 
94 @param[in] usart unsigned 32 bit. USART block register address base @ref
95 usart_reg_base
96 @param[in] bits unsigned 32 bit. Word length in bits 8 or 9.
97 */
98 
99 void usart_set_databits(uint32_t usart, uint32_t bits)
100 {
101  if (bits == 8) {
102  USART_CR1(usart) &= ~USART_CR1_M; /* 8 data bits */
103  } else {
104  USART_CR1(usart) |= USART_CR1_M; /* 9 data bits */
105  }
106 }

109 /** @brief USART Set Stop Bit(s).
110 
111 The stop bits are specified as 0.5, 1, 1.5 or 2.
112 
113 @param[in] usart unsigned 32 bit. USART block register address base @ref
114 usart_reg_base
115 @param[in] stopbits unsigned 32 bit. Stop bits @ref usart_cr2_stopbits.
116 */
117 
118 void usart_set_stopbits(uint32_t usart, uint32_t stopbits)
119 {
120  uint32_t reg32;
121 
122  reg32 = USART_CR2(usart);
123  reg32 = (reg32 & ~USART_CR2_STOPBITS_MASK) | stopbits;
124  USART_CR2(usart) = reg32;
125 }
