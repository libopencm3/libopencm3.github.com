22/**
23 * \brief Enable the clock source for the peripheral
24 *
25 * @param[in] periph peripheral and clock type to enable @see lm4f_clken
26 */
27void periph_clock_enable(enum lm4f_clken periph)
28{
29 MMIO32(SYSCTL_BASE + (periph >> 5)) |= 1 << (periph & 0x1f);
30}
31
32/**
33 * \brief Disable the clock source for the peripheral
34 *
35 * @param[in] periph peripheral and clock type to enable @see lm4f_clken
36 */
37void periph_clock_disable(enum lm4f_clken periph)
38{
39 MMIO32(SYSCTL_BASE + (periph >> 5)) &= ~(1 << (periph & 0x1f));
40}
