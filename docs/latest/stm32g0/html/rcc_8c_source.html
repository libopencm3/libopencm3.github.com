128void rcc_osc_on(enum rcc_osc osc)
129{
130 switch (osc) {
131 case RCC_PLL:
132 RCC_CR |= RCC_CR_PLLON;
133 break;
134 case RCC_HSE:
135 RCC_CR |= RCC_CR_HSEON;
136 break;
137 case RCC_HSI:
138 RCC_CR |= RCC_CR_HSION;
139 break;
140 case RCC_LSE:
141 RCC_BDCR |= RCC_BDCR_LSEON;
142 break;
143 case RCC_LSI:
144 RCC_CSR |= RCC_CSR_LSION;
145 break;
146 default:
147 cm3_assert_not_reached();
148 break;
149 }
150}
151
152void rcc_osc_off(enum rcc_osc osc)
153{
154 switch (osc) {
155 case RCC_PLL:
156 RCC_CR &= ~RCC_CR_PLLON;
157 break;
158 case RCC_HSE:
159 RCC_CR &= ~RCC_CR_HSEON;
160 break;
161 case RCC_HSI:
162 RCC_CR &= ~RCC_CR_HSION;
163 break;
164 case RCC_LSE:
165 RCC_BDCR &= ~RCC_BDCR_LSEON;
166 break;
167 case RCC_LSI:
168 RCC_CSR &= ~RCC_CSR_LSION;
169 break;
170 default:
171 cm3_assert_not_reached();
172 break;
173 }
174}

176bool rcc_is_osc_ready(enum rcc_osc osc)
177{
178 switch (osc) {
179 case RCC_PLL:
180 return RCC_CR & RCC_CR_PLLRDY;
181 case RCC_HSE:
182 return RCC_CR & RCC_CR_HSERDY;
183 case RCC_HSI:
184 return RCC_CR & RCC_CR_HSIRDY;
185 case RCC_LSE:
186 return RCC_BDCR & RCC_BDCR_LSERDY;
187 case RCC_LSI:
188 return RCC_CSR & RCC_CSR_LSIRDY;
189 default:
190 cm3_assert_not_reached();
191 return 0;
192 }
193 return false;
194}

196void rcc_wait_for_osc_ready(enum rcc_osc osc)
197{
198 while (!rcc_is_osc_ready(osc));

222/** @brief Set the Source for the System Clock.
223 * @param osc Oscillator to use.
224 */
225void rcc_set_sysclk_source(enum rcc_osc osc)
226{
227 uint32_t reg32;
228 uint32_t sw = 0;
229
230 switch (osc) {
231 case RCC_HSI:
232 sw = RCC_CFGR_SW_HSISYS;
233 break;
234 case RCC_HSE:
235 sw = RCC_CFGR_SW_HSE;
236 break;
237 case RCC_PLL:
238 sw = RCC_CFGR_SW_PLLRCLK;
239 break;
240 case RCC_LSE:
241 sw = RCC_CFGR_SW_LSE;
242 break;
243 case RCC_LSI:
244 sw = RCC_CFGR_SW_LSI;
245 break;
246 default:
247 cm3_assert_not_reached();
248 return;
249 }
250
251 reg32 = RCC_CFGR;
252 reg32 &= ~(RCC_CFGR_SW_MASK << RCC_CFGR_SW_SHIFT);
253 RCC_CFGR = (reg32 | (sw << RCC_CFGR_SW_SHIFT));
254}

267 case RCC_CFGR_SWS_PLLRCLK:
268 return RCC_PLL;
269 case RCC_CFGR_SW_LSE:
270 return RCC_LSE;

274 cm3_assert_not_reached();
275 return 0;

280/** @brief Wait until system clock switched to given oscillator.
281 * @param osc Oscillator.
282 */
283void rcc_wait_for_sysclk_status(enum rcc_osc osc)
284{
285 uint32_t sws = 0;
286
287 switch (osc) {
288 case RCC_PLL:
289 sws = RCC_CFGR_SWS_PLLRCLK;
290 break;
291 case RCC_HSE:
292 sws = RCC_CFGR_SWS_HSE;
293 break;
294 case RCC_HSI:
295 sws = RCC_CFGR_SWS_HSISYS;
296 break;
297 case RCC_LSI:
298 sws = RCC_CFGR_SWS_LSI;
299 break;
300 case RCC_LSE:
301 sws = RCC_CFGR_SWS_LSE;
302 break;
303 default:
304 cm3_assert_not_reached();
305 break;
306 }
307
308 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) != sws);
309}

435/**
436 * @brief Setup sysclock with desired source (HSE/HSI/PLL/LSE/LSI). taking care of flash/pwr and src configuration
437 * @param clock rcc_clock_scale with desired parameters
438 */
439void rcc_clock_setup(const struct rcc_clock_scale *clock)
440{
441 if (clock->sysclock_source == RCC_PLL) {
442 enum rcc_osc pll_source;
443
444 if (clock->pll_source == RCC_PLLCFGR_PLLSRC_HSE)
445 pll_source = RCC_HSE;
446 else
447 pll_source = RCC_HSI;
448
449 /* start pll src osc. */
450 rcc_osc_on(pll_source);
451 rcc_wait_for_osc_ready(pll_source);
452
453 /* stop pll to reconfigure it. */
454 rcc_osc_off(RCC_PLL);
455 while (rcc_is_osc_ready(RCC_PLL));
456
457 rcc_set_main_pll(clock->pll_source, clock->pll_div, clock->pll_mul, clock->pllp_div, clock->pllq_div, clock->pllr_div);
458
459 rcc_enable_pllr(true);
460 } else if (clock->sysclock_source == RCC_HSI) {
461 rcc_set_hsisys_div(clock->hsisys_div);
462 }
463
464 rcc_periph_clock_enable(RCC_PWR);
465 pwr_set_vos_scale(clock->voltage_scale);
466
467 flash_set_ws(clock->flash_waitstates);
468
469 /* enable flash prefetch if we have at least 1WS */
470 if (clock->flash_waitstates > FLASH_ACR_LATENCY_0WS)
471 flash_prefetch_enable();
472 else
473 flash_prefetch_disable();
474
475 rcc_set_hpre(clock->hpre);
476 rcc_set_ppre(clock->ppre);
477
478 rcc_osc_on(clock->sysclock_source);
479 rcc_wait_for_osc_ready(clock->sysclock_source);
480
481 rcc_set_sysclk_source(clock->sysclock_source);
482 rcc_wait_for_sysclk_status(clock->sysclock_source);
483
484 rcc_ahb_frequency = clock->ahb_frequency;
485 rcc_apb1_frequency = clock->apb_frequency;
486}

498/**
499 * @brief Set the peripheral clock source
500 * @param periph peripheral of choice, eg XXX_BASE
501 * @param sel periphral clock source
502 */
503void rcc_set_peripheral_clk_sel(uint32_t periph, uint32_t sel)
504{
505 uint8_t shift;
506 uint32_t mask;
507
508 switch (periph) {
509 case ADC1_BASE:
510 shift = RCC_CCIPR_ADCSEL_SHIFT;
511 mask = RCC_CCIPR_ADCSEL_MASK;
512 break;
513 case RNG_BASE:
514 shift = RCC_CCIPR_RNGSEL_SHIFT;
515 mask = RCC_CCIPR_RNGSEL_MASK;
516 break;
517 case TIM1_BASE:
518 shift = RCC_CCIPR_TIM1SEL_SHIFT;
519 mask = RCC_CCIPR_TIM1SEL_MASK;
520 break;
521 case LPTIM1_BASE:
522 shift = RCC_CCIPR_LPTIM1SEL_SHIFT;
523 mask = RCC_CCIPR_LPTIM1SEL_MASK;
524 break;
525 case LPTIM2_BASE:
526 shift = RCC_CCIPR_LPTIM2SEL_SHIFT;
527 mask = RCC_CCIPR_LPTIM2SEL_MASK;
528 break;
529 case CEC_BASE:
530 shift = RCC_CCIPR_CECSEL_SHIFT;
531 mask = RCC_CCIPR_CECSEL_MASK;
532 break;
533 case USART2_BASE:
534 shift = RCC_CCIPR_USART2SEL_SHIFT;
535 mask = RCC_CCIPR_USARTxSEL_MASK;
536 break;
537 case USART1_BASE:
538 shift = RCC_CCIPR_USART1SEL_SHIFT;
539 mask = RCC_CCIPR_USARTxSEL_MASK;
540 break;
541 default:
542 cm3_assert_not_reached();
543 return;
544 }
545
546 uint32_t reg32 = RCC_CCIPR & ~(mask << shift);
547 RCC_CCIPR = reg32 | (sel << shift);
548}

550static uint32_t rcc_get_clksel_freq(uint8_t shift) {
551 uint8_t clksel = (RCC_CCIPR >> shift) & RCC_CCIPR_USARTxSEL_MASK;
552 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
553 switch (clksel) {
554 case RCC_CCIPR_USARTxSEL_PCLK:
555 return rcc_apb1_frequency;
556 case RCC_CCIPR_USARTxSEL_SYSCLK:
557 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
558 case RCC_CCIPR_USARTxSEL_LSE:
559 return 32768;
560 case RCC_CCIPR_USARTxSEL_HSI16:
561 return 16000000U;
562 }
563 cm3_assert_not_reached();
564}
