163bool rcc_is_osc_ready(enum rcc_osc osc)
164{
165 switch (osc) {

168 case RCC_HSE:

170 case RCC_HSI:

172 case RCC_LSE:

185void rcc_wait_for_sysclk_status(enum rcc_osc osc)
186{
187 switch (osc) {
188 case RCC_PLL:
189 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
190 RCC_CFGR_SWS_PLL);
191 break;
192 case RCC_HSE:
193 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
194 RCC_CFGR_SWS_HSE);
195 break;
196 case RCC_HSI:
197 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
198 RCC_CFGR_SWS_HSI);
199 break;
200 default:
201 /* Shouldn't be reached. */
202 break;
203 }
204}

312void rcc_set_main_pll_hsi(uint32_t pllm, uint32_t plln, uint32_t pllp,
313 uint32_t pllq)
314{
315 RCC_PLLCFGR = (pllm << RCC_PLLCFGR_PLLM_SHIFT) |
316 (plln << RCC_PLLCFGR_PLLN_SHIFT) |

318 (pllq << RCC_PLLCFGR_PLLQ_SHIFT);

337void rcc_clock_setup_hse_3v3(const struct rcc_clock_scale *clock)
338{
339 /* Enable internal high-speed oscillator. */
340 rcc_osc_on(RCC_HSI);
341 rcc_wait_for_osc_ready(RCC_HSI);
342
343 /* Select HSI as SYSCLK source. */
344 rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
345
346 /* Enable external high-speed oscillator 8MHz. */
347 rcc_osc_on(RCC_HSE);
348 rcc_wait_for_osc_ready(RCC_HSE);
349
350 /*
351 * Set prescalers for AHB, ADC, APB1, APB2.
352 * Do this before touching the PLL (TODO: why?).
353 */
354 rcc_set_hpre(clock->hpre);
355 rcc_set_ppre1(clock->ppre1);
356 rcc_set_ppre2(clock->ppre2);
357
358 /* Disable PLL oscillator before changing its configuration. */
359 rcc_osc_off(RCC_PLL);
360
361 /* Configure the PLL oscillator. */
362 rcc_set_main_pll_hse(clock->pllm, clock->plln,
363 clock->pllp, clock->pllq);
364
365 /* Enable PLL oscillator and wait for it to stabilize. */
366 rcc_osc_on(RCC_PLL);
367 rcc_wait_for_osc_ready(RCC_PLL);
368
369 /* Configure flash settings. */
370 flash_set_ws(clock->flash_config);
371
372 /* Select PLL as SYSCLK source. */
373 rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
374
375 /* Wait for PLL clock to be selected. */
376 rcc_wait_for_sysclk_status(RCC_PLL);
377
378 /* Set the peripheral clock frequencies used. */
379 rcc_apb1_frequency = clock->apb1_frequency;
380 rcc_apb2_frequency = clock->apb2_frequency;
381}

406/*---------------------------------------------------------------------------*/
407/** @brief Get the peripheral clock speed for the Timer at base specified.
408 * @param timer Base address of TIM to get clock frequency for.
409 */
410uint32_t rcc_get_timer_clk_freq(uint32_t timer)
411{
412 /* Handle APB1 timer clocks. */
413 if (timer >= TIM2_BASE && timer <= TIM14_BASE) {
414 uint8_t ppre1 = (RCC_CFGR >> RCC_CFGR_PPRE1_SHIFT) & RCC_CFGR_PPRE1_MASK;
415 return (ppre1 == RCC_CFGR_PPRE_DIV_NONE) ? rcc_apb1_frequency
416 : 2 * rcc_apb1_frequency;
417 } else {
418 uint8_t ppre2 = (RCC_CFGR >> RCC_CFGR_PPRE2_SHIFT) & RCC_CFGR_PPRE2_MASK;
419 return (ppre2 == RCC_CFGR_PPRE_DIV_NONE) ? rcc_apb2_frequency
420 : 2 * rcc_apb2_frequency;
421 }
422}
