53/** Initialize the USB device controller hardware of the STM32. */
54static usbd_device *stm32f107_usbd_init(void)
55{
56 rcc_periph_clock_enable(RCC_OTGFS);
57 OTG_FS_GUSBCFG |= OTG_GUSBCFG_PHYSEL;
58
59 /* Wait for AHB idle. */
60 while (!(OTG_FS_GRSTCTL & OTG_GRSTCTL_AHBIDL));
61 /* Do core soft reset. */
62 OTG_FS_GRSTCTL |= OTG_GRSTCTL_CSRST;
63 while (OTG_FS_GRSTCTL & OTG_GRSTCTL_CSRST);
64
65 if (OTG_FS_CID >= OTG_CID_HAS_VBDEN) {
66 /* Enable VBUS detection in device mode and power up the PHY. */
67 OTG_FS_GCCFG |= OTG_GCCFG_VBDEN | OTG_GCCFG_PWRDWN;
68 } else {
69 /* Enable VBUS sensing in device mode and power up the PHY. */
70 OTG_FS_GCCFG |= OTG_GCCFG_VBUSBSEN | OTG_GCCFG_PWRDWN;
71 }
72 /* Explicitly enable DP pullup (not all cores do this by default) */
73 OTG_FS_DCTL &= ~OTG_DCTL_SDIS;
74
75 /* Force peripheral only mode. */
76 OTG_FS_GUSBCFG |= OTG_GUSBCFG_FDMOD | OTG_GUSBCFG_TRDT_MASK;
77
78 OTG_FS_GINTSTS = OTG_GINTSTS_MMIS;
79
80 /* Full speed device. */
81 OTG_FS_DCFG |= OTG_DCFG_DSPD;
82
83 /* Restart the PHY clock. */
84 OTG_FS_PCGCCTL = 0;
85
86 OTG_FS_GRXFSIZ = stm32f107_usb_driver.rx_fifo_size;
87 usbd_dev.fifo_mem_top = stm32f107_usb_driver.rx_fifo_size;
88
89 /* Unmask interrupts for TX and RX. */
90 OTG_FS_GAHBCFG |= OTG_GAHBCFG_GINT;
91 OTG_FS_GINTMSK = OTG_GINTMSK_ENUMDNEM |
92 OTG_GINTMSK_RXFLVLM |
93 OTG_GINTMSK_IEPINT |
94 OTG_GINTMSK_USBSUSPM |
95 OTG_GINTMSK_WUIM;
96 OTG_FS_DAINTMSK = 0xF;
97 OTG_FS_DIEPMSK = OTG_DIEPMSK_XFRCM;
98
99 return &usbd_dev;
100}
