32 /** @brief Set RTC prescalars.
33 
34 This sets the RTC synchronous and asynchronous prescalars.
35 */
36 
37 void rtc_set_prescaler(uint32_t sync, uint32_t async)
38 {
39  /*
40  * Even if only one of the two fields needs to be changed,
41  * 2 separate write accesses must be performed to the RTC_PRER register.
42  */
43  RTC_PRER = (sync & RTC_PRER_PREDIV_S_MASK);
44  RTC_PRER |= (async << RTC_PRER_PREDIV_A_SHIFT);
45 }

48 /** @brief Wait for RTC registers to be synchronised with the APB1 bus
49 
50  Time and Date are accessed through shadow registers which must be synchronized
51 */
52 
53 void rtc_wait_for_synchro(void)
54 {
55  /* Unlock RTC registers */
56  RTC_WPR = 0xca;
57  RTC_WPR = 0x53;
58 
59  RTC_ISR &= ~(RTC_ISR_RSF);

62 
63  /* disable write protection again */
64  RTC_WPR = 0xff;

87 /** @brief Sets the wakeup time auto-reload value
88 
89 */
90 void rtc_set_wakeup_time(uint16_t wkup_time, uint8_t rtc_cr_wucksel)
91 {
92  /* FTFM:
93  * The following sequence is required to configure or change the wakeup
94  * timer auto-reload value (WUT[15:0] in RTC_WUTR):
95  * 1. Clear WUTE in RTC_CR to disable the wakeup timer.
96  */
97  RTC_CR &= ~RTC_CR_WUTE;
98  /* 2. Poll WUTWF until it is set in RTC_ISR to make sure the access to
99  * wakeup auto-reload counter and to WUCKSEL[2:0] bits is allowed.
100  * It takes around 2 RTCCLK clock cycles (due to clock
101  * synchronization).
102  */
103  while (!((RTC_ISR) & (RTC_ISR_WUTWF)));
104  /* 3. Program the wakeup auto-reload value WUT[15:0], and the wakeup
105  * clock selection (WUCKSEL[2:0] bits in RTC_CR).Set WUTE in RTC_CR
106  * to enable the timer again. The wakeup timer restarts
107  * down-counting.
108  */
109  RTC_WUTR = wkup_time;
110  RTC_CR &= ~(RTC_CR_WUCLKSEL_MASK << RTC_CR_WUCLKSEL_SHIFT);
111  RTC_CR |= (rtc_cr_wucksel << RTC_CR_WUCLKSEL_SHIFT);
112  RTC_CR |= RTC_CR_WUTE;
113 }
