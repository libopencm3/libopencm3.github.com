36void flash_set_ws(uint32_t ws)
37{
38 uint32_t reg32;
39
40 reg32 = FLASH_ACR;
41 reg32 &= ~(FLASH_ACR_LATENCY_MASK << FLASH_ACR_LATENCY_SHIFT);
42 reg32 |= (ws << FLASH_ACR_LATENCY_SHIFT);
43 FLASH_ACR = reg32;
44}
