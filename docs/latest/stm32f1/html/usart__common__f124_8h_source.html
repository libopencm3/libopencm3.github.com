110/** @defgroup usart_convenience_flags U(S)ART convenience Flags
111 * @ingroup STM32F_usart_defines

113 * allowing them to be used regardless of which usart peripheral
114 * version you have.
115 * @{
116 */

118#define USART_FLAG_FE USART_SR_FE

121#define USART_FLAG_IDLE USART_SR_IDLE

127/* --- USART_SR values ----------------------------------------------------- */
128/****************************************************************************/
129/** @defgroup usart_sr_flags USART Status register Flags
130@ingroup STM32F_usart_defines
131
132@{*/
133
134/** CTS: CTS flag */
135/** @note: undefined on UART4 and UART5 */
136#define USART_SR_CTS (1 << 9)
137
138/** LBD: LIN break detection flag */
139#define USART_SR_LBD (1 << 8)
140
141/** TXE: Transmit data buffer empty */
142#define USART_SR_TXE (1 << 7)
143
144/** TC: Transmission complete */
145#define USART_SR_TC (1 << 6)
146
147/** RXNE: Read data register not empty */
148#define USART_SR_RXNE (1 << 5)
149
150/** IDLE: Idle line detected */
151#define USART_SR_IDLE (1 << 4)
152
153/** ORE: Overrun error */
154#define USART_SR_ORE (1 << 3)
155
156/** NE: Noise error flag */
157#define USART_SR_NE (1 << 2)
158
159/** FE: Framing error */
160#define USART_SR_FE (1 << 1)
161
162/** PE: Parity error */
163#define USART_SR_PE (1 << 0)
164/**@}*/
