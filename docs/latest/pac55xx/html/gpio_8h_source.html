258/**
259 * Set the IO mode and pull-up/down configuration for the pins.
260 * @param[in] gpioport Port to configure the alternate function on.
261 * @param[in] mode IO Mode to configure (analog, input, output).
262 * @param[in] pull_up_down Pull configuration (up/down/none) to set for the pins..
263 * @param[in] gpios Pins to set with the mode and pull config specified.
264 */
265void gpio_mode_setup(uint32_t gpioport, gpio_mode_t mode,
266 ccs_pull_updown_t pull_up_down, uint16_t gpios);
267/**
268 * Set the output mask (disable changes) to the output pins specified.
269 * @param[in] gpioport Port to set output mask on.
270 * @param[in] enable Whether to enable or disable output masking.
271 * @param[in] gpios bitfield of pins to set.
272 */
273void gpio_set_outmask(uint32_t gpioport, bool enable, uint16_t gpios);
274/**
275 * Set the port pins specified to the true state.
276 * @param[in] gpioport Port to set bits on.
277 * @param[in] gpios bitfield of pins to set.
278 */
279void gpio_set(uint32_t gpioport, uint16_t gpios);
280/**
281 * Clear the port pins specified to the false state.
282 * @param[in] gpioport Port to clear bits on.
283 * @param[in] gpios bitfield of pins to clear.
284 */
285void gpio_clear(uint32_t gpioport, uint16_t gpios);
286/**
287 * Return a masked bitfield of the port specified.
288 * @param[in] gpioport Port to read the bits from.
289 * @param[in] gpios bitfield mask to apply to the port read.
290 * @return masked bitfield of the port.
291 */
292uint16_t gpio_get(uint32_t gpioport, uint16_t gpios);
293/**
294 * Set the function of the pin for this port. This will modify the pinmux,
295 * @param[in] gpioport Port to configure the alternate function on.
296 * @param[in] muxsel Mux select mode to configure on the port and pins.
297 * @param[in] gpios Pins to set with the function specified.
298 */
299void gpio_set_af(uint32_t gpioport, ccs_muxsel_func_t muxsel, uint16_t gpios);
300
301/**
302 * Set special output options for the gpio pin. For this MCU, this is only the drive strength.
303 * @param[in] gpioport Port to configure the alternate function on.
304 * @param[in] strength Drive strength (DS_XXMA from ccs.h).
305 * @param[in] gpios Pins to set with the drive strength specified.
306 */
307void gpio_set_output_options(uint32_t gpioport, ccs_drive_strength_t strength,
308 uint16_t gpios);
309/**
310 * Set input schmidt trigger for glitch rejection on the input pin.
311 * @param[in] gpioport Port to configure the alternate function on.
312 * @param[in] enable True to enable, false to disable the schmidt trigger.
313 * @param[in] gpios Pins to set with the schmidt trigger setting specified.
314 */
315void gpio_set_schmidt_trigger(uint32_t gpioport, bool enable, uint16_t gpios);
