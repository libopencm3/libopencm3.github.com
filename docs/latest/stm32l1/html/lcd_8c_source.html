103void lcd_set_refresh_frequency(uint32_t frequency)
104{
105 uint32_t duty, lcd_clock;
106 switch ((LCD_CR >> LCD_CR_DUTY_SHIFT) & LCD_CR_DUTY_MASK) {
107 case LCD_CR_DUTY_STATIC:
108 duty = 1;
109 break;
110 case LCD_CR_DUTY_1_2:
111 duty = 2;
112 break;
113 case LCD_CR_DUTY_1_3:
114 duty = 3;
115 break;
116 case LCD_CR_DUTY_1_4:
117 duty = 4;
118 break;
119 case LCD_CR_DUTY_1_8:
120 duty = 8;
121 break;
122 default:
123 /* Incorrect duty */
124 return;
125 }
126
127 switch ((RCC_CSR >> RCC_CSR_RTCSEL_SHIFT) & RCC_CSR_RTCSEL_MASK) {
128 case RCC_CSR_RTCSEL_LSE:
129 lcd_clock = 32786;
130 break;
131 case RCC_CSR_RTCSEL_LSI:
132 lcd_clock = 37000;
133 break;
134 case RCC_CSR_RTCSEL_HSE:
135 /* no current method of determining clock and divider! */
136 return;
137 default:
138 /* RCC Clock not selected */
139 return;
140 }
141
142 /* PS * DIV = lcd_clock/(duty * freq) */
143 uint32_t ps_mul_div = lcd_clock / (duty * frequency);
144
145 int div, ps = 0;
146 while (ps_mul_div > 32) {
147 ps_mul_div >>= 1;
148 ps++;
149 }
150 div = ps_mul_div - 16;
151
152 lcd_set_prescaler(ps);
153 lcd_set_divider(div);
154}
