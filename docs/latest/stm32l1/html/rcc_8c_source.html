256 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
257 {
258  switch (osc) {
259  case RCC_PLL:
260  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
261  RCC_CFGR_SWS_SYSCLKSEL_PLLCLK);
262  break;
263  case RCC_HSE:
264  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
265  RCC_CFGR_SWS_SYSCLKSEL_HSECLK);
266  break;
267  case RCC_HSI:
268  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
269  RCC_CFGR_SWS_SYSCLKSEL_HSICLK);
270  break;
271  case RCC_MSI:
272  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
273  RCC_CFGR_SWS_SYSCLKSEL_MSICLK);
274  break;
275  default:
276  /* Shouldn't be reached. */
277  break;
278  }

465 /**
466  * Switch sysclock to HSI with the given parameters.
467  * This should be usable from any point in time, but only if you have used
468  * library functions to manage clocks. It relies on the global
469  * @ref rcc_ahb_frequency to ensure that it reliably scales voltage up or down
470  * as appropriate.
471  * @param clock full struct with desired parameters
472  */
473 void rcc_clock_setup_hsi(const struct rcc_clock_scale *clock)
474 {
475  /* Enable internal high-speed oscillator. */
476  rcc_osc_on(RCC_HSI);
477  rcc_periph_clock_enable(RCC_PWR);
478 
479  /* I guess this should be in the settings? */
480  flash_64bit_enable();
481  flash_prefetch_enable();
482 
483  /* Don't try and go to fast for a voltage range! */
484  if (clock->ahb_frequency > rcc_ahb_frequency) {
485  /* Going up, power up first */
486  pwr_set_vos_scale(clock->voltage_scale);
487  rcc_set_hpre(clock->hpre);
488  rcc_set_ppre1(clock->ppre1);
489  rcc_set_ppre2(clock->ppre2);
490  flash_set_ws(clock->flash_waitstates);
491  } else {
492  /* going down, slow down before cutting power */
493  rcc_set_hpre(clock->hpre);
494  rcc_set_ppre1(clock->ppre1);
495  rcc_set_ppre2(clock->ppre2);
496  flash_set_ws(clock->flash_waitstates);
497  pwr_set_vos_scale(clock->voltage_scale);
498  }
499 
500  rcc_wait_for_osc_ready(RCC_HSI);
501  while (PWR_CSR & PWR_CSR_VOSF) {
502  ;
503  }
504  rcc_set_sysclk_source(RCC_CFGR_SW_SYSCLKSEL_HSICLK);
505 
506  /* Set the peripheral clock frequencies used. */
507  rcc_ahb_frequency = clock->ahb_frequency;
508  rcc_apb1_frequency = clock->apb1_frequency;
509  rcc_apb2_frequency = clock->apb2_frequency;
510 }

512 void rcc_clock_setup_pll(const struct rcc_clock_scale *clock)
513 {
514  /* Turn on the appropriate source for the PLL */
515  if (clock->pll_source == RCC_CFGR_PLLSRC_HSE_CLK) {
516  rcc_osc_on(RCC_HSE);
517  rcc_wait_for_osc_ready(RCC_HSE);
518  } else {
519  rcc_osc_on(RCC_HSI);
520  rcc_wait_for_osc_ready(RCC_HSI);
521  }
522 
523  /*
524  * Set prescalers for AHB, ADC, APB1, APB2.
525  * Do this before touching the PLL (TODO: why?).
526  */
527  rcc_set_hpre(clock->hpre);
528  rcc_set_ppre1(clock->ppre1);
529  rcc_set_ppre2(clock->ppre2);
530 
531  rcc_periph_clock_enable(RCC_PWR);
532  pwr_set_vos_scale(clock->voltage_scale);
533 
534  /* I guess this should be in the settings? */
535  flash_64bit_enable();
536  flash_prefetch_enable();
537  flash_set_ws(clock->flash_waitstates);
538 
539  /* Disable PLL oscillator before changing its configuration. */
540  rcc_osc_off(RCC_PLL);
541 
542  /* Configure the PLL oscillator. */
543  rcc_set_pll_configuration(clock->pll_source, clock->pll_mul,
544  clock->pll_div);

547  rcc_osc_on(RCC_PLL);
548  rcc_wait_for_osc_ready(RCC_PLL);

551  rcc_set_sysclk_source(RCC_CFGR_SW_SYSCLKSEL_PLLCLK);
552 
553  /* Set the peripheral clock frequencies used. */
554  rcc_ahb_frequency = clock->ahb_frequency;
555  rcc_apb1_frequency = clock->apb1_frequency;
556  rcc_apb2_frequency = clock->apb2_frequency;

574  * @param timer Base address of TIM to get clock frequency for.
575  */
576 uint32_t rcc_get_timer_clk_freq(uint32_t timer)
577 {
578  /* Handle APB1 timers, and apply multiplier if necessary. */
579  if (timer >= TIM2_BASE && timer <= TIM7_BASE) {
580  uint8_t ppre1 = (RCC_CFGR >> RCC_CFGR_PPRE1_SHIFT) & RCC_CFGR_PPRE1_MASK;
581  return (ppre1 == RCC_CFGR_PPRE1_HCLK_NODIV) ? rcc_apb1_frequency
582  : 2 * rcc_apb1_frequency;
583  } else {
584  uint8_t ppre2 = (RCC_CFGR >> RCC_CFGR_PPRE2_SHIFT) & RCC_CFGR_PPRE2_MASK;
585  return (ppre2 == RCC_CFGR_PPRE2_HCLK_NODIV) ? rcc_apb2_frequency
586  : 2 * rcc_apb2_frequency;
587  }
