155/*
156 * Some config bits only count in CR1 or TR1, even if you
157 * are just configuring bank 2, so these masks let you copy
158 * out those bits after you have computed values for CR2 and
159 * TR2 and put them into CR1 and TR1
160 */
161#define FMC_SDTR_DNC_MASK (FMC_SDTR_TRP_MASK | FMC_SDTR_TRC_MASK)
162#define FMC_SDCR_DNC_MASK (FMC_SDCR_SDCLK_MASK | \
163 FMC_SDCR_RPIPE_MASK | \
164 FMC_SDCR_RBURST)
