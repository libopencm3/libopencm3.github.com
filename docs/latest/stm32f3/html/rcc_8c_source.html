184bool rcc_is_osc_ready(enum rcc_osc osc)
185{
186 switch (osc) {
187 case RCC_PLL:
188 return RCC_CR & RCC_CR_PLLRDY;
189 case RCC_HSE:
190 return RCC_CR & RCC_CR_HSERDY;
191 case RCC_HSI:
192 return RCC_CR & RCC_CR_HSIRDY;
193 case RCC_LSE:
194 return RCC_BDCR & RCC_BDCR_LSERDY;
195 case RCC_LSI:
196 return RCC_CSR & RCC_CSR_LSIRDY;
197 }
198 return false;
199}

212void rcc_wait_for_sysclk_status(enum rcc_osc osc)
213{
214 switch (osc) {
215 case RCC_PLL:
216 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
217 RCC_CFGR_SWS_PLL);
218 break;
219 case RCC_HSE:
220 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
221 RCC_CFGR_SWS_HSE);
222 break;
223 case RCC_HSI:
224 while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
225 RCC_CFGR_SWS_HSI);
226 break;
227 default:
228 /* Shouldn't be reached. */
229 break;
230 }
231}

357/**
358 * Setup clocks to run from PLL.
359 * The arguments provide the pll source, multipliers, dividers, all that's
360 * needed to establish a system clock.
361 * @param clock clock information structure
362 */
363void rcc_clock_setup_pll(const struct rcc_clock_scale *clock)
364{
365 if (clock->pllsrc == RCC_CFGR_PLLSRC_HSE_PREDIV) {
366 rcc_osc_on(RCC_HSE);
367 rcc_wait_for_osc_ready(RCC_HSE);
368 } else {
369 rcc_osc_on(RCC_HSI);
370 rcc_wait_for_osc_ready(RCC_HSI);
371 }
372 rcc_osc_off(RCC_PLL);
373 rcc_usb_prescale_1_5();
374 if (clock->usbdiv1) {
375 rcc_usb_prescale_1();
376 }
377 rcc_wait_for_osc_not_ready(RCC_PLL);
378 rcc_set_pll_source(clock->pllsrc);
379 rcc_set_pll_multiplier(clock->pllmul);
380 rcc_set_prediv(clock->plldiv);
381 /* Enable PLL oscillator and wait for it to stabilize. */
382 rcc_osc_on(RCC_PLL);
383 rcc_wait_for_osc_ready(RCC_PLL);
384
385 /* Configure flash settings. */
386 flash_prefetch_enable();
387 flash_set_ws(clock->flash_waitstates);
388
389 rcc_set_hpre(clock->hpre);
390 rcc_set_ppre2(clock->ppre2);
391 rcc_set_ppre1(clock->ppre1);
392 /* Select PLL as SYSCLK source. */
393 rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
394 /* Wait for PLL clock to be selected. */
395 rcc_wait_for_sysclk_status(RCC_PLL);
396
397 /* Set the peripheral clock frequencies used. */
398 rcc_ahb_frequency = clock->ahb_frequency;
399 rcc_apb1_frequency = clock->apb1_frequency;
400 rcc_apb2_frequency = clock->apb2_frequency;
401}

485void rcc_adc_prescale(uint32_t prescale1, uint32_t prescale2)
486{
487 uint32_t clear_mask = (RCC_CFGR2_ADCxPRES_MASK
488 << RCC_CFGR2_ADC12PRES_SHIFT)
489 | (RCC_CFGR2_ADCxPRES_MASK
490 << RCC_CFGR2_ADC34PRES_SHIFT);
491 uint32_t set = (prescale1 << RCC_CFGR2_ADC12PRES_SHIFT) |
492 (prescale2 << RCC_CFGR2_ADC34PRES_SHIFT);
493 RCC_CFGR2 &= ~(clear_mask);
494 RCC_CFGR2 |= (set);
495}

497static uint32_t rcc_get_usart_clksel_freq(uint32_t apb_clk, uint8_t shift) {
498 uint8_t clksel = (RCC_CFGR3 >> shift) & RCC_CFGR3_UARTxSW_MASK;
499 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
500 switch (clksel) {
501 case RCC_CFGR3_UARTxSW_PCLK:
502 return apb_clk;
503 case RCC_CFGR3_UARTxSW_SYSCLK:
504 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
505 case RCC_CFGR3_UARTxSW_LSE:
506 return 32768;
507 case RCC_CFGR3_UARTxSW_HSI:
508 return 8000000U;
509 }
510 cm3_assert_not_reached();
511}

514/** @brief Get the peripheral clock speed for the USART at base specified.
515 * @param usart Base address of USART to get clock frequency for.
516 */
517uint32_t rcc_get_usart_clk_freq(uint32_t usart)
518{
519 /* Handle values with selectable clocks. */
520 if (usart == USART1_BASE) {
521 return rcc_get_usart_clksel_freq(rcc_apb2_frequency, RCC_CFGR3_UART1SW_SHIFT);
522 } else if (usart == USART2_BASE) {
523 return rcc_get_usart_clksel_freq(rcc_apb1_frequency, RCC_CFGR3_UART2SW_SHIFT);
524 } else if (usart == USART3_BASE) {
525 return rcc_get_usart_clksel_freq(rcc_apb1_frequency, RCC_CFGR3_UART3SW_SHIFT);
526 } else if (usart == UART4_BASE) {
527 return rcc_get_usart_clksel_freq(rcc_apb1_frequency, RCC_CFGR3_UART4SW_SHIFT);
528 } else { /* UART5 */
529 return rcc_get_usart_clksel_freq(rcc_apb1_frequency, RCC_CFGR3_UART5SW_SHIFT);
530 }
531}

534/** @brief Get the peripheral clock speed for the Timer at base specified.
535 * @param timer Base address of TIM to get clock frequency for.
536 */
537uint32_t rcc_get_timer_clk_freq(uint32_t timer)
538{
539 /* Handle APB1 timer clocks. */
540 if (timer >= TIM2_BASE && timer <= TIM7_BASE) {
541 uint8_t ppre1 = (RCC_CFGR >> RCC_CFGR_PPRE1_SHIFT) & RCC_CFGR_PPRE1_MASK;
542 return (ppre1 == RCC_CFGR_PPRE1_DIV_NONE) ? rcc_apb1_frequency
543 : 2 * rcc_apb1_frequency;
544 } else {
545 uint8_t ppre2 = (RCC_CFGR >> RCC_CFGR_PPRE2_SHIFT) & RCC_CFGR_PPRE2_MASK;
546 return (ppre2 == RCC_CFGR_PPRE2_DIV_NONE) ? rcc_apb2_frequency
547 : 2 * rcc_apb2_frequency;
548 }
549}

552/** @brief Get the peripheral clock speed for the I2C device at base specified.
553 * @param i2c Base address of I2C to get clock frequency for.
554 */
555uint32_t rcc_get_i2c_clk_freq(uint32_t i2c)
556{
557 if (i2c == I2C1_BASE) {
558 if (RCC_CFGR3 & RCC_CFGR3_I2C1SW) {
559 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
560 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
561 } else {
562 return 8000000U;
563 }
564 } else {
565 return rcc_apb1_frequency;
566 }
567}
