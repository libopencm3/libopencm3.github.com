212 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
213 {
214  switch (osc) {
215  case RCC_PLL:
216  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
217  RCC_CFGR_SWS_PLL);
218  break;
219  case RCC_HSE:
220  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
221  RCC_CFGR_SWS_HSE);
222  break;
223  case RCC_HSI:
224  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK) !=
225  RCC_CFGR_SWS_HSI);
226  break;
227  default:
228  /* Shouldn't be reached. */
229  break;
230  }

357 /**
358  * Setup clocks to run from PLL.
359  * The arguments provide the pll source, multipliers, dividers, all that's
360  * needed to establish a system clock.
361  * @param clock clock information structure
362  */
363 void rcc_clock_setup_pll(const struct rcc_clock_scale *clock)
364 {
365  if (clock->pllsrc == RCC_CFGR_PLLSRC_HSE_PREDIV) {
366  rcc_osc_on(RCC_HSE);
367  rcc_wait_for_osc_ready(RCC_HSE);
368  } else {
369  rcc_osc_on(RCC_HSI);
370  rcc_wait_for_osc_ready(RCC_HSI);
371  }
372  rcc_osc_off(RCC_PLL);
373  rcc_usb_prescale_1_5();
374  if (clock->usbdiv1) {
375  rcc_usb_prescale_1();
376  }
377  rcc_wait_for_osc_not_ready(RCC_PLL);
378  rcc_set_pll_source(clock->pllsrc);
379  rcc_set_pll_multiplier(clock->pllmul);
380  rcc_set_prediv(clock->plldiv);
381  /* Enable PLL oscillator and wait for it to stabilize. */
382  rcc_osc_on(RCC_PLL);
383  rcc_wait_for_osc_ready(RCC_PLL);
384 
385  /* Configure flash settings. */
386  flash_prefetch_enable();
387  flash_set_ws(clock->flash_waitstates);
388 
389  rcc_set_hpre(clock->hpre);
390  rcc_set_ppre2(clock->ppre2);
391  rcc_set_ppre1(clock->ppre1);
392  /* Select PLL as SYSCLK source. */
393  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
394  /* Wait for PLL clock to be selected. */
395  rcc_wait_for_sysclk_status(RCC_PLL);
396 
397  /* Set the peripheral clock frequencies used. */
398  rcc_ahb_frequency = clock->ahb_frequency;
399  rcc_apb1_frequency = clock->apb1_frequency;
400  rcc_apb2_frequency = clock->apb2_frequency;
401 }

497 static uint32_t rcc_get_usart_clksel_freq(uint32_t apb_clk, uint8_t shift) {
498  uint8_t clksel = (RCC_CFGR3 >> shift) & RCC_CFGR3_UARTxSW_MASK;
499  uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
500  switch (clksel) {
501  case RCC_CFGR3_UART1SW_PCLK:
502  return apb_clk;
503  case RCC_CFGR3_UART1SW_SYSCLK:
504  return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
505  case RCC_CFGR3_UART1SW_HSI:
506  return 8000000U;
507  }
508  cm3_assert_not_reached();
509 }

535 uint32_t rcc_get_timer_clk_freq(uint32_t timer)

539  uint8_t ppre1 = (RCC_CFGR >> RCC_CFGR_PPRE1_SHIFT) & RCC_CFGR_PPRE1_MASK;
540  return (ppre1 == RCC_CFGR_PPRE1_DIV_NONE) ? rcc_apb1_frequency
541  : 2 * rcc_apb1_frequency;
542  } else {
543  uint8_t ppre2 = (RCC_CFGR >> RCC_CFGR_PPRE2_SHIFT) & RCC_CFGR_PPRE2_MASK;
544  return (ppre2 == RCC_CFGR_PPRE2_DIV_NONE) ? rcc_apb2_frequency
545  : 2 * rcc_apb2_frequency;
546  }
