49/** @brief RCC Clear the Oscillator Ready Interrupt Flag
50 *
51 * Clear the interrupt flag that was set when a clock oscillator became ready
52 * to use.
53 *
54 * @param osc Oscillator ID
55 */
56
57void rcc_osc_ready_int_clear(enum rcc_osc osc)
58{
59 switch (osc) {
60 case RCC_HSI48:
61 RCC_CIR |= RCC_CIR_HSI48RDYC;
62 break;
63 case RCC_HSI14:
64 RCC_CIR |= RCC_CIR_HSI14RDYC;
65 break;
66 case RCC_HSI:
67 RCC_CIR |= RCC_CIR_HSIRDYC;
68 break;
69 case RCC_HSE:
70 RCC_CIR |= RCC_CIR_HSERDYC;
71 break;
72 case RCC_PLL:
73 RCC_CIR |= RCC_CIR_PLLRDYC;
74 break;
75 case RCC_LSE:
76 RCC_CIR |= RCC_CIR_LSERDYC;
77 break;
78 case RCC_LSI:
79 RCC_CIR |= RCC_CIR_LSIRDYC;
80 break;
81 }
82}

206bool rcc_is_osc_ready(enum rcc_osc osc)
207{
208 switch (osc) {
209 case RCC_HSI48:
210 return RCC_CR2 & RCC_CR2_HSI48RDY;
211 case RCC_HSI14:
212 return RCC_CR2 & RCC_CR2_HSI14RDY;
213 case RCC_HSI:
214 return RCC_CR & RCC_CR_HSIRDY;
215 case RCC_HSE:
216 return RCC_CR & RCC_CR_HSERDY;
217 case RCC_PLL:
218 return RCC_CR & RCC_CR_PLLRDY;
219 case RCC_LSE:
220 return RCC_BDCR & RCC_BDCR_LSERDY;
221 case RCC_LSI:
222 return RCC_CSR & RCC_CSR_LSIRDY;
223 }
224 return false;
225}

227void rcc_wait_for_osc_ready(enum rcc_osc osc)
228{
229 while (!rcc_is_osc_ready(osc));

232/*---------------------------------------------------------------------------*/
233/** @brief RCC Turn on an Oscillator.
234 *
235 * Enable an oscillator and power on. Each oscillator requires an amount of
236 * time to settle to a usable state. Refer to datasheets for time delay
237 * information. A status flag is available to indicate when the oscillator
238 * becomes ready (see @ref rcc_osc_ready_int_flag and @ref
239 * rcc_wait_for_osc_ready).
240 *
241 * @param osc Oscillator ID
242 */
243
244void rcc_osc_on(enum rcc_osc osc)
245{
246 switch (osc) {
247 case RCC_HSI48:
248 RCC_CR2 |= RCC_CR2_HSI48ON;
249 break;
250 case RCC_HSI14:
251 RCC_CR2 |= RCC_CR2_HSI14ON;
252 break;
253 case RCC_HSI:
254 RCC_CR |= RCC_CR_HSION;
255 break;
256 case RCC_HSE:
257 RCC_CR |= RCC_CR_HSEON;
258 break;
259 case RCC_LSE:
260 RCC_BDCR |= RCC_BDCR_LSEON;
261 break;
262 case RCC_LSI:
263 RCC_CSR |= RCC_CSR_LSION;
264 break;
265 case RCC_PLL:
266 RCC_CR |= RCC_CR_PLLON;
267 break;
268 }
269}

271/*---------------------------------------------------------------------------*/
272/** @brief RCC Turn off an Oscillator.
273 *
274 * Disable an oscillator and power off.
275 *
276 * @note An oscillator cannot be turned off if it is selected as the system
277 * clock.
278 *
279 * @param osc Oscillator ID

281
282void rcc_osc_off(enum rcc_osc osc)
283{
284 switch (osc) {

288 case RCC_HSI14:

290 break;
291 case RCC_HSI:
292 RCC_CR &= ~RCC_CR_HSION;
293 break;
294 case RCC_HSE:
295 RCC_CR &= ~RCC_CR_HSEON;
296 break;
297 case RCC_LSE:
298 RCC_BDCR &= ~RCC_BDCR_LSEON;
299 break;
300 case RCC_LSI:
301 RCC_CSR &= ~RCC_CSR_LSION;
302 break;
303 case RCC_PLL:
304 /* don't do anything */
305 break;
306 }
307}

554/**
555 * Set System Clock PLL at 48MHz from HSE at 8MHz.
556 */
557void rcc_clock_setup_in_hse_8mhz_out_48mhz(void)
558{
559 rcc_osc_on(RCC_HSE);
560 rcc_wait_for_osc_ready(RCC_HSE);
561 rcc_set_sysclk_source(RCC_HSE);
562
563 rcc_set_hpre(RCC_CFGR_HPRE_NODIV);
564 rcc_set_ppre(RCC_CFGR_PPRE_NODIV);
565
566 flash_prefetch_enable();
567 flash_set_ws(FLASH_ACR_LATENCY_024_048MHZ);
568
569 /* PLL: 8MHz * 6 = 48MHz */
570 rcc_set_pll_multiplication_factor(RCC_CFGR_PLLMUL_MUL6);
571 rcc_set_pll_source(RCC_CFGR_PLLSRC_HSE_CLK);
572 rcc_set_pllxtpre(RCC_CFGR_PLLXTPRE_HSE_CLK);
573
574 rcc_osc_on(RCC_PLL);
575 rcc_wait_for_osc_ready(RCC_PLL);
576 rcc_set_sysclk_source(RCC_PLL);
577
578 rcc_apb1_frequency = 48000000;
579 rcc_ahb_frequency = 48000000;
580}

629static uint32_t rcc_get_usart_clksel_freq(uint8_t shift) {
630 uint8_t clksel = (RCC_CFGR3 >> shift) & RCC_CFGR3_USARTxSW_MASK;
631 uint8_t hpre = (RCC_CFGR >> RCC_CFGR_HPRE_SHIFT) & RCC_CFGR_HPRE_MASK;
632 switch (clksel) {
633 case RCC_CFGR3_USARTxSW_PCLK:
634 return rcc_apb1_frequency;
635 case RCC_CFGR3_USARTxSW_SYSCLK:
636 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
637 case RCC_CFGR3_USARTxSW_LSE:
638 return 32768;
639 case RCC_CFGR3_USARTxSW_HSI:
640 return 8000000U;
641 }
642 cm3_assert_not_reached();
643}

646/** @brief Get the peripheral clock speed for the USART at base specified.
647 * @param usart Base address of USART to get clock frequency for.
648 */
649uint32_t rcc_get_usart_clk_freq(uint32_t usart)
650{
651 if (usart == USART1_BASE) {
652 return rcc_get_usart_clksel_freq(RCC_CFGR3_USART1SW_SHIFT);
653 } else if (usart == USART2_BASE) {
654 return rcc_get_usart_clksel_freq(RCC_CFGR3_USART2SW_SHIFT);
655 } else if (usart == USART3_BASE) {
656 return rcc_get_usart_clksel_freq(RCC_CFGR3_USART3SW_SHIFT);
657 } else {
658 return rcc_apb1_frequency;
659 }
660}

663/** @brief Get the peripheral clock speed for the Timer at base specified.
664 * @param timer Base address of TIM to get clock frequency for.
665 */
666uint32_t rcc_get_timer_clk_freq(uint32_t timer __attribute__((unused)))
667{
668 uint8_t ppre = (RCC_CFGR >> RCC_CFGR_PPRE_SHIFT) & RCC_CFGR_PPRE_MASK;
669 return (ppre == RCC_CFGR_PPRE_NODIV) ? rcc_apb1_frequency
670 : 2 * rcc_apb1_frequency;
671}

674/** @brief Get the peripheral clock speed for the I2C device at base specified.
675 * @param i2c Base address of I2C to get clock frequency for.
676 */
677uint32_t rcc_get_i2c_clk_freq(uint32_t i2c)
678{

682 return rcc_ahb_frequency * rcc_get_div_from_hpre(hpre);
683 } else {
684 return 8000000U;
685 }
686 } else {
687 return rcc_apb1_frequency;
688 }
689}
